972 template <typename triangulationType>
973 int ttk::MandatoryCriticalPoints::buildSubTrees(
974  const triangulationType &triangulation) {
975 
976  Timer t;
977 
978 #ifndef TTK_ENABLE_KAMIKAZE
979  if(vertexNumber_ <= 0)
980  return -1;
981  if((int)upperVertexScalars_.size() != vertexNumber_)
982  return -2;
983  if((int)lowerVertexScalars_.size() != vertexNumber_)
984  return -3;
985  if((int)vertexPositions_.size() != vertexNumber_)
986  return -4;
987  if(triangulation.getNumberOfVertices() != vertexNumber_)
988  return -5;
989  if((int)vertexSoSoffsets_.size() != vertexNumber_)
990  return -6;
991 #endif
992 
993  // upperMaximumList_ and lowerMinimumList_ computation (not sorted by function
994  // value)
995  lowerMinimumList_.clear();
996  upperMaximumList_.clear();
997 #ifdef TTK_ENABLE_OPENMP
998 #pragma omp parallel for num_threads(threadNumber_)
999 #endif
1000  for(int i = 0; i < vertexNumber_; i++) {
1001  bool isLowerMin = true;
1002  bool isUpperMax = true;
1003  SimplexId neighborNumber = triangulation.getVertexNeighborNumber(i);
1004  for(SimplexId j = 0; j < neighborNumber; j++) {
1005  SimplexId neighborId;
1006  triangulation.getVertexNeighbor(i, j, neighborId);
1007  if((lowerVertexScalars_[neighborId] < lowerVertexScalars_[i])
1008  || ((lowerVertexScalars_[neighborId] == lowerVertexScalars_[i])
1009  && (vertexSoSoffsets_[neighborId] < vertexSoSoffsets_[i])))
1010  isLowerMin = false;
1011  if((upperVertexScalars_[neighborId] > upperVertexScalars_[i])
1012  || ((upperVertexScalars_[neighborId] == upperVertexScalars_[i])
1013  && (vertexSoSoffsets_[neighborId] > vertexSoSoffsets_[i])))
1014  isUpperMax = false;
1015  if(!isUpperMax && !isLowerMin)
1016  break;
1017  }
1018  if(isLowerMin) {
1019 #ifdef TTK_ENABLE_OPENMP
1020 #pragma omp critical
1021 #endif
1022  { lowerMinimumList_.push_back(i); }
1023  }
1024  if(isUpperMax) {
1025 #ifdef TTK_ENABLE_OPENMP
1026 #pragma omp critical
1027 #endif
1028  { upperMaximumList_.push_back(i); }
1029  }
1030  }
1031 
1032 #ifdef TTK_ENABLE_OPENMP
1033 #pragma omp parallel sections num_threads(threadNumber_)
1034 #endif
1035  {
1036 #ifdef TTK_ENABLE_OPENMP
1037 #pragma omp section
1038 #endif
1039  {
1040  upperJoinTree_.setNumberOfVertices(vertexNumber_);
1041  upperJoinTree_.setVertexScalars(&upperVertexScalars_);
1042  upperJoinTree_.setVertexPositions(&vertexPositions_);
1043  upperJoinTree_.setTriangulation(&triangulation);
1044  upperJoinTree_.setVertexSoSoffsets(&vertexSoSoffsets_);
1045  upperJoinTree_.buildExtremumList(upperMinimumList_, true);
1046  upperJoinTree_.build();
1047  }
1048 #ifdef TTK_ENABLE_OPENMP
1049 #pragma omp section
1050 #endif
1051  {
1052  lowerJoinTree_.setNumberOfVertices(vertexNumber_);
1053  lowerJoinTree_.setVertexScalars(&lowerVertexScalars_);
1054  lowerJoinTree_.setVertexPositions(&vertexPositions_);
1055  lowerJoinTree_.setTriangulation(&triangulation);
1056  lowerJoinTree_.setVertexSoSoffsets(&vertexSoSoffsets_);
1057  lowerJoinTree_.setMinimumList(lowerMinimumList_);
1058  lowerJoinTree_.build();
1059  }
1060 #ifdef TTK_ENABLE_OPENMP
1061 #pragma omp section
1062 #endif
1063  {
1064  upperSplitTree_.setNumberOfVertices(vertexNumber_);
1065  upperSplitTree_.setVertexScalars(&upperVertexScalars_);
1066  upperSplitTree_.setVertexPositions(&vertexPositions_);
1067  upperSplitTree_.setTriangulation(&triangulation);
1068  upperSplitTree_.setVertexSoSoffsets(&vertexSoSoffsets_);
1069  upperSplitTree_.setMaximumList(upperMaximumList_);
1070  upperSplitTree_.build();
1071  }
1072 #ifdef TTK_ENABLE_OPENMP
1073 #pragma omp section
1074 #endif
1075  {

1077  lowerSplitTree_.setVertexScalars(&lowerVertexScalars_);
1078  lowerSplitTree_.setVertexPositions(&vertexPositions_);
1079  lowerSplitTree_.setTriangulation(&triangulation);
1080  lowerSplitTree_.setVertexSoSoffsets(&vertexSoSoffsets_);
1081  lowerSplitTree_.buildExtremumList(lowerMaximumList_, false);
1082  lowerSplitTree_.build();
1083  }
1084  }
1085  this->printMsg("4 SubLevelSetTrees computed", 1.0, t.getElapsedTime(),
1086  this->threadNumber_);
1087  return 0;

1091 int ttk::MandatoryCriticalPoints::computeSaddleComponent(
1092  const int componentId,
1093  const PointType &pointType,
1094  const std::vector<std::pair<int, int>> &mandatorySaddleVertex,
1095  const std::vector<double> &lowVertexScalars,
1096  const std::vector<double> &upVertexScalars,
1097  std::vector<int> &componentVertexList,
1098  const triangulationType &triangulation) const {
1099 
1100  const int seedVertexId = mandatorySaddleVertex[componentId].first;
1101  const double lowInterval = lowVertexScalars[seedVertexId];
1102  const double upInterval
1103  = upVertexScalars[mandatorySaddleVertex[componentId].second];
1104 
1105  componentVertexList.clear();
1106 
1107  std::vector<bool> isVisited(vertexNumber_, false);
1108  std::queue<SimplexId> idQueue;
1109  idQueue.push(seedVertexId);
1110 
1111  while(!(idQueue.empty())) {
1112  int vertexId = idQueue.front();
1113  idQueue.pop();
1114  if(!isVisited[vertexId]) {
1115  isVisited[vertexId] = true;
1116  double lowerValue = lowerVertexScalars_[vertexId];
1117  double upperValue = upperVertexScalars_[vertexId];
1118  if((pointType == PointType::JoinSaddle && (!(lowerValue > upInterval))
1119  && (upperValue > lowInterval))
1120  || (pointType == PointType::SplitSaddle
1121  && (!(upperValue < lowInterval)) && (lowerValue < upInterval))) {
1122  componentVertexList.push_back(vertexId);
1123  // Neighbors
1124  SimplexId neighborNumber
1125  = triangulation.getVertexNeighborNumber(vertexId);
1126  for(SimplexId i = 0; i < neighborNumber; i++) {
1127  SimplexId neighborVertexId;
1128  triangulation.getVertexNeighbor(vertexId, i, neighborVertexId);
1129  idQueue.push(neighborVertexId);
1130  }
1131  }
1132  }
1133  }
1134  return 0;
1135 }
