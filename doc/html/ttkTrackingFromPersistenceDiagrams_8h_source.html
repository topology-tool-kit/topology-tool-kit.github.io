618 
619  vtkIntArray *extremumIndexScalars = vtkIntArray::SafeDownCast(

623  CTPersistenceDiagram_->GetCellData()->GetArray("Persistence"));

626  CTPersistenceDiagram_->GetPointData()->GetArray("Birth"));
627

629  CTPersistenceDiagram_->GetPointData()->GetArray("Death"));
630 
631  vtkPoints *points = (CTPersistenceDiagram_->GetPoints());
632  if(!pairIdentifierScalars)
633  return -2;
634 
635  auto pairingsSize = (int)pairIdentifierScalars->GetNumberOfTuples();
636 
637  // Continuous indexing (no gap in indices)
638  for(int pairIndex = 0; pairIndex < pairingsSize; ++pairIndex) {
639  const float indexOfPair = pairIndex;
640  if(*pairIdentifierScalars->GetTuple(pairIndex) != -1) // except diagonal
641  pairIdentifierScalars->SetTuple(pairIndex, &indexOfPair);
642  }
643 
644  auto s = (float)0.0;
645 
646  if(!deathScalars != !birthScalars)
647  return -2;
648  bool is2D = !deathScalars && !birthScalars;
649  bool is3D = !is2D;
650  if(Is3D && !is3D)
651  Is3D = false;
652  if(!Is3D && diagramNumber == 1)
653  s = (float)spacing;
654 
655  if(pairingsSize < 1 || !vertexIdentifierScalars || !nodeTypeScalars
656  || !persistenceScalars || !extremumIndexScalars || !points)
657  return -2;
658 
659  diagram.resize((unsigned long)pairingsSize);
660  int nbNonCompact = 0;
661 
662  for(int i = 0; i < pairingsSize; ++i) {
663 
664  int vertexId1 = vertexIdentifierScalars->GetValue(2 * i);
665  int vertexId2 = vertexIdentifierScalars->GetValue(2 * i + 1);
666  int nodeType1 = nodeTypeScalars->GetValue(2 * i);
667  int nodeType2 = nodeTypeScalars->GetValue(2 * i + 1);
668 
669  int pairIdentifier = pairIdentifierScalars->GetValue(i);
670  int pairType = extremumIndexScalars->GetValue(i);
671  double persistence = persistenceScalars->GetValue(i);
672 
673  int index1 = 2 * i;
674  double *coords1 = points->GetPoint(index1);
675  auto x1 = (float)coords1[0];
676  auto y1 = (float)coords1[1];
677  auto z1 = (float)coords1[2];
678 
679  int index2 = index1 + 1;
680  double *coords2 = points->GetPoint(index2);
681  auto x2 = (float)coords2[0];
682  auto y2 = (float)coords2[1];
683  auto z2 = (float)coords2[2];
684 
685  dataType value1 = (!birthScalars) ? (dataType)x1
686  : (dataType)birthScalars->GetValue(2 * i);
687  dataType value2 = (!deathScalars)
688  ? (dataType)y2
689  : (dataType)deathScalars->GetValue(2 * i + 1);
690 
691  if(pairIdentifier != -1 && pairIdentifier < pairingsSize)
692  diagram.at(pairIdentifier)
693  = std::make_tuple(vertexId1, (BNodeType)nodeType1, vertexId2,
694  (BNodeType)nodeType2, (dataType)persistence, pairType,
695  value1, x1, y1, z1 + s, value2, x2, y2, z2 + s);
696 
697  if(pairIdentifier >= pairingsSize) {
698  nbNonCompact++;
699  if(nbNonCompact == 0) {
700  std::stringstream msg;
701  msg << "[TTKBottleneckDistance] Diagram pair identifiers "
702  << "must be compact (not exceed the diagram size). " << std::endl;
703  dMsg(std::cout, msg.str(), timeMsg);
704  }
705  }
706  }
707 
708  if(nbNonCompact > 0) {
709  {
710  std::stringstream msg;
711  msg << "[TTKBottleneckDistance] Missed " << nbNonCompact
712  << " pairs due to non-compactness." << std::endl;
713  dMsg(std::cout, msg.str(), timeMsg);
714  }
715  }
716 
717  sort(diagram.begin(), diagram.end(),
718  [](const diagramTuple &a, const diagramTuple &b) -> bool {
719  return std::get<6>(a) < std::get<6>(b);
720  });
721 
722  return 0;
723 }
