916 }

921 template <typename T, typename U>
922 MergeTree(const T scalarsT, U paramsT)
923 : scalars(scalarsT), params(paramsT),
924 tree(paramsT, scalarsT, params->treeType) {
925 tree.makeAlloc();
926 scalarsValues = std::make_shared<std::vector<dataType>>();
927 for(unsigned int i = 0; i < tree.getNumberOfNodes(); ++i)
928 scalarsValues->push_back(tree.getValue<dataType>(i));
929 scalars->values = (void *)(scalarsValues->data());
930 }
931
932 MergeTree(const std::shared_ptr<ftm::Scalars> &scalarsT,
933 const std::shared_ptr<std::vector<dataType>> &scalarValuesT,
934 std::shared_ptr<ftm::Params> &paramsT)
935 : scalars(scalarsT), scalarsValues(scalarValuesT), params(paramsT),
936 tree(paramsT, scalarsT, params->treeType) {
937 tree.makeAlloc();
938 scalars->values = (void *)(scalarsValues->data());
939 }
940
941 void copy(const MergeTree<dataType> &mt) {
942 // Copy scalars
943 scalars = std::make_shared<ftm::Scalars>();
944 scalars->size = mt.scalars->size;
945 scalarsValues = mt.scalarsValues;
946 scalars->values = (void *)(scalarsValues->data());
947
948 // Copy params
949 params = std::make_shared<ftm::Params>();
950 params->treeType = mt.params->treeType;
951
952 // Copy tree
953 tree.clear();
954 tree.setParamsScalars(params, scalars);
955 tree.makeAlloc();
956 tree.copyMergeTreeStructure(const_cast<FTMTree_MT *>(&(mt.tree)));
957 }
958
959 MergeTree(const MergeTree<dataType> &mt)
960 : scalars(mt.scalars), scalarsValues(mt.scalarsValues),
961 params(mt.params), tree(params, scalars, params->treeType) {
962 copy(mt);
963 }
964
965 MergeTree<dataType> &operator=(const MergeTree<dataType> &mt) {
966 if(&mt != this) {
967 copy(mt);
968 }
969 return *this;
970 }
971 };
