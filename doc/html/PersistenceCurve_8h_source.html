111template <typename scalarType>
112int ttk::PersistenceCurve::computePersistencePlot(
113 const std::vector<std::tuple<SimplexId, SimplexId, scalarType>> &pairs,
114 std::vector<std::pair<scalarType, SimplexId>> &plot) const {
115
116 SimplexId nbElmnt = pairs.size();
117 plot.resize(nbElmnt);
118
119 // build curve
120 const scalarType epsilon
121 = static_cast<scalarType>(Geometry::powIntTen(-REAL_SIGNIFICANT_DIGITS));
122 for(SimplexId i = 0; i < nbElmnt; ++i) {
123 plot[i].first = std::max(std::get<2>(pairs[i]), epsilon);
124 plot[i].second = pairs.size() - i;
125 }
126
127 return 0;
128}

131int ttk::PersistenceCurve::execute(const scalarType *inputScalars,
132 const size_t scalarsMTime,
133 const SimplexId *inputOffsets,
134 const triangulationType *triangulation) {
135
136 printMsg(ttk::debug::Separator::L1);
137
138 Timer timer;
139
140 using plotType = std::vector<std::pair<scalarType, SimplexId>>;
141
142 auto JTPlot = static_cast<plotType *>(JTPlot_);
143 auto STPlot = static_cast<plotType *>(STPlot_);
144 auto CTPlot = static_cast<plotType *>(CTPlot_);
145 auto MSCPlot = static_cast<plotType *>(MSCPlot_);
146
147 contourTree_.setVertexScalars(inputScalars);
148 contourTree_.setTreeType(ftm::TreeType::Join_Split);
149 contourTree_.setVertexSoSoffsets(inputOffsets);
150 contourTree_.setSegmentation(false);
151 contourTree_.setThreadNumber(threadNumber_);
152 contourTree_.build<scalarType>(triangulation);
153
154 // get persistence pairs
155 std::vector<std::tuple<SimplexId, SimplexId, scalarType>> JTPairs;
156 std::vector<std::tuple<SimplexId, SimplexId, scalarType>> STPairs;
157 contourTree_.computePersistencePairs<scalarType>(JTPairs, true);
158 contourTree_.computePersistencePairs<scalarType>(STPairs, false);
159
160 // merge pairs
161 std::vector<std::tuple<SimplexId, SimplexId, scalarType>> CTPairs(
162 JTPairs.size() + STPairs.size());
163 std::copy(JTPairs.begin(), JTPairs.end(), CTPairs.begin());
164 std::copy(STPairs.begin(), STPairs.end(), CTPairs.begin() + JTPairs.size());
165 {
166 auto cmp = [](const std::tuple<SimplexId, SimplexId, scalarType> &a,
167 const std::tuple<SimplexId, SimplexId, scalarType> &b) {
168 return std::get<2>(a) < std::get<2>(b);
169 };
170 std::sort(CTPairs.begin(), CTPairs.end(), cmp);
171 }
172
173 // get the saddle-saddle pairs
174 const int dimensionality = triangulation->getDimensionality();
175 if(dimensionality == 3 and ComputeSaddleConnectors and MSCPlot_ != nullptr) {
176 std::vector<std::tuple<SimplexId, SimplexId, scalarType>>
177 pl_saddleSaddlePairs;
178 dcg_.setInputScalarField(inputScalars, scalarsMTime);
179 dcg_.setInputOffsets(inputOffsets);
180 dcg_.computeSaddleSaddlePersistencePairs<scalarType>(
181 pl_saddleSaddlePairs, *triangulation);
182
183 // sort the saddle-saddle pairs by persistence value and compute curve
184 {
185 auto cmp = [](const std::tuple<SimplexId, SimplexId, scalarType> &a,
186 const std::tuple<SimplexId, SimplexId, scalarType> &b) {
187 return std::get<2>(a) < std::get<2>(b);
188 };
189 std::sort(pl_saddleSaddlePairs.begin(), pl_saddleSaddlePairs.end(), cmp);
190 }
191
192 computePersistencePlot<scalarType>(pl_saddleSaddlePairs, *MSCPlot);
193 }
194
195 // get persistence curves
196#ifdef TTK_ENABLE_OPENMP
197#pragma omp parallel sections
198#endif
199 {
200#ifdef TTK_ENABLE_OPENMP
201#pragma omp section
202#endif
203 if(JTPlot_ != nullptr) {
204 computePersistencePlot<scalarType>(JTPairs, *JTPlot);
205 }
206#ifdef TTK_ENABLE_OPENMP
207#pragma omp section
208#endif
209 if(STPlot_ != nullptr) {
210 computePersistencePlot<scalarType>(STPairs, *STPlot);
211 }
212#ifdef TTK_ENABLE_OPENMP
213#pragma omp section
214#endif
215 if(CTPlot_ != nullptr) {
216 computePersistencePlot<scalarType>(CTPairs, *CTPlot);
217 }
218 }
219
220 printMsg(
221 "Base execution completed", 1, timer.getElapsedTime(), threadNumber_);
222 printMsg(ttk::debug::Separator::L1);
223
224 return 0;
