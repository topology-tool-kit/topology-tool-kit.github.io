347template <typename triangulationType>
348ttk::SimplexId ttk::QuadrangulationSubdivision::findEdgeMiddle(
349 const size_t a,
350 const size_t b,
351 const triangulationType &triangulation) const {
352
353 std::vector<SimplexId> midId(this->threadNumber_);
354 std::vector<float> minValue(
355 this->threadNumber_, std::numeric_limits<float>::infinity());
356
357 // euclidian barycenter of a and b
358 Point edgeEuclBary = (outputPoints_[a] + outputPoints_[b]) * 0.5F;
359
360#ifdef TTK_ENABLE_OPENMP
361#pragma omp parallel for num_threads(threadNumber_)
362#endif // TTK_ENABLE_OPENMP
363 for(size_t i = 0; i < vertexDistance_[a].size(); ++i) {
364#ifdef TTK_ENABLE_OPENMP
365 const auto tid = omp_get_thread_num();
366#else
367 const auto tid = 0;
368#endif // TTK_ENABLE_OPENMP
369 float m = vertexDistance_[a][i];
370 float n = vertexDistance_[b][i];
371 // stay on the shortest path between a and b
372 float sum = m + n;
373
374 // skip further computation
375 if(sum > minValue[tid]) {
376 continue;
377 }
378
379 if(m != std::numeric_limits<float>::infinity()
380 && n != std::numeric_limits<float>::infinity()) {
381 // try to get the middle of the shortest path
382 sum += std::abs(m - n);
383 }
384
385 // get the euclidian distance to AB
386 Point curr{};
387 triangulation.getVertexPoint(i, curr.x, curr.y, curr.z);
388 // try to minimize the euclidian distance to AB too
389 sum += Geometry::distance(&curr.x, &edgeEuclBary.x);
390
391 // search for the minimizing index
392 if(sum < minValue[tid]) {
393 minValue[tid] = sum;
394 midId[tid] = i;
395 }
396 }
397
398#ifdef TTK_ENABLE_OPENMP
399 for(int i = 1; i < this->threadNumber_; ++i) {
400 if(minValue[i] < minValue[0]) {
401 minValue[0] = minValue[i];
402 midId[0] = midId[i];
403 }
404 }
405#endif // TTK_ENABLE_OPENMP
406
407 return midId[0];
408}

410template <typename triangulationType>
411std::tuple<ttk::QuadrangulationSubdivision::Point,
412 ttk::SimplexId,
413 size_t,
414 ttk::SimplexId>
415 ttk::QuadrangulationSubdivision::findProjection(
416 const size_t a,
417 const bool forceReverseProj,
418 const triangulationType &triangulation) const {
419
420 static const float PREC_FLT{powf(10, -FLT_DIG)};
421
422 // current vertex 3d coordinates
423 Point pa = outputPoints_[a];
424
425 Point res{};
426
427 // fallback to euclidian projection code if no normals
428 bool doReverseProj = forceReverseProj;
429
430 // quad normal in a
431 Point normalsMean{};
432
433 if(doReverseProj) {
434 // compute mean of normals
435 normalsMean = getQuadNormal(a);
436
437 doReverseProj = !std::isnan(normalsMean.x);
438 }
439
440 // found a projection in one triangle
441 bool success = false;
442 // list of triangle IDs to test to find a potential projection
443 std::stack<SimplexId> trianglesToTest;
444 // list of triangle IDs already tested
445 // (takes more memory to reduce computation time)
446 std::vector<bool> trianglesTested(
447 triangulation.getNumberOfTriangles(), false);
448 // number of triangles tested
449 size_t trChecked{0};
450 // vertex in triangle with highest barycentric coordinate
451 SimplexId nearestVertex = nearestVertexIdentifier_[a];
452
453 // number of triangles around nearest vertex
454 SimplexId triangleNumber
455 = triangulation.getVertexTriangleNumber(nearestVertex);
456 // init pipeline by checking in every triangle around selected vertex
457 for(SimplexId j = 0; j < triangleNumber; j++) {
458 SimplexId ntid;
459 triangulation.getVertexTriangle(nearestVertex, j, ntid);
460 trianglesToTest.push(ntid);
461 }
462
463 while(!trianglesToTest.empty()) {
464 SimplexId i = trianglesToTest.top();
465 trianglesToTest.pop();
466
467 // skip if already tested
468 if(trianglesTested[i]) {
469 continue;
470 }
471
472 // get triangle vertices
473 std::array<SimplexId, 3> tverts{};
474 triangulation.getTriangleVertex(i, 0, tverts[0]);
475 triangulation.getTriangleVertex(i, 1, tverts[1]);
476 triangulation.getTriangleVertex(i, 2, tverts[2]);
477
478 // get coordinates of triangle vertices
479 Point pm{}, pn{}, po{};
480 triangulation.getVertexPoint(tverts[0], pm.x, pm.y, pm.z);
481 triangulation.getVertexPoint(tverts[1], pn.x, pn.y, pn.z);
482 triangulation.getVertexPoint(tverts[2], po.x, po.y, po.z);
483
484 // triangle normal: cross product of two edges
485 Point crossP{};
486 // mn, mo vectors
487 Point mn = pn - pm;
488 Point mo = po - pm;
489 // compute mn ^ mo
490 Geometry::crossProduct(&mn.x, &mo.x, &crossP.x);
491 // unitary normal vector
492 Point normTri = crossP / Geometry::magnitude(&crossP.x);
493
494 // compute intersection of triangle plane and line (a, normalsMean)
495 if(doReverseProj) {
496
497 auto denom = Geometry::dotProduct(&normalsMean.x, &normTri.x);
498
499 // check if triangle plane is parallel to quad normal
500 if(std::abs(denom) < PREC_FLT) {
501 // skip this iteration after filling pipeline
502 trianglesTested[i] = true;
503 // fill pipeline with neighboring triangles
504 for(auto &vert : tverts) {
505 auto ntr = triangulation.getVertexTriangleNumber(vert);
506 for(SimplexId j = 0; j < ntr; ++j) {
507 SimplexId tid;
508 triangulation.getVertexTriangle(vert, j, tid);
509 if(tid != i) {
510 trianglesToTest.push(tid);
511 }
512 }
513 }
514 continue;
515 }
516
517 // use formula from Wikipedia: line-plane intersection
518 auto tmp = pm - pa;
519 auto alpha = Geometry::dotProduct(&tmp.x, &normTri.x) / denom;
520
521 // intersection
522 res = pa + normalsMean * alpha;
523
524 }
525 // compute euclidian projection of a in triangle plane
526 else {
527
528 auto tmp = pa - pm;
529 // projection
530 res = pa - normTri * Geometry::dotProduct(&normTri.x, &tmp.x);
531 }
532
533 // compute barycentric coords of projection
534 std::array<float, 3> baryCoords{};
535 Geometry::computeBarycentricCoordinates(
536 &pm.x, &pn.x, &po.x, &res.x, baryCoords);
537
538 // check if projection in triangle
539 bool inTriangle = true;
540 for(auto &coord : baryCoords) {
541 if(coord < PREC_FLT) {
542 inTriangle = false;
543 }
544 if(coord > 1 + PREC_FLT) {
545 inTriangle = false;
546 }
547 }
548
549 // mark triangle as tested
550 trianglesTested[i] = true;
551 trChecked++;
552
553 if(inTriangle) {
554 success = true;
555 // should we check if we have the nearest triangle?
556 break;
557 }
558
559 // extrema values in baryCoords
560 auto extrema = std::minmax_element(baryCoords.begin(), baryCoords.end());
561
562 // find the nearest triangle vertices (with the highest/positive
563 // values in baryCoords) from proj
564 std::vector<SimplexId> vertices(2);
565 vertices[0] = tverts[extrema.second - baryCoords.begin()];
566 for(size_t j = 0; j < baryCoords.size(); j++) {
567 if(j != static_cast<size_t>(extrema.first - baryCoords.begin())
568 && j != static_cast<size_t>(extrema.second - baryCoords.begin())) {
569 vertices[1] = tverts[j];
570 break;
571 }
572 }
573
574 // store vertex with highest barycentric coordinate
575 nearestVertex = vertices[0];
576
577 // triangles around vertices[0] and vertices[1]
578 std::array<std::set<SimplexId>, 2> vertsTriangles{};
579
580 // get triangles around vertices
581 for(size_t j = 0; j < vertices.size(); ++j) {
582 SimplexId tnum = triangulation.getVertexTriangleNumber(vertices[j]);
583 for(SimplexId k = 0; k < tnum; k++) {
584 SimplexId tid;
585 triangulation.getVertexTriangle(vertices[j], k, tid);
586 if(tid == i) {
587 continue;
588 }
589 vertsTriangles[j].insert(tid);
590 }
591 }
592
593 // triangles to test next
594 std::vector<SimplexId> common_triangles;
595
596 // look for triangles sharing the vertices with max values in baryCoords
597 std::set_intersection(vertsTriangles[0].begin(), vertsTriangles[0].end(),
598 vertsTriangles[1].begin(), vertsTriangles[1].end(),
599 std::back_inserter(common_triangles));
600
601 for(auto &ntid : common_triangles) {
602 if(!trianglesTested[ntid]) {
603 trianglesToTest.push(ntid);
604 }
605 }
606 }
607
608 const size_t maxTrChecked = 100;
609
610 if(success && trChecked > maxTrChecked) {
611 success = false;
612 }
613
614 if(!success) {
615 if(!forceReverseProj) {
616 return findProjection(a, true, triangulation);
617 }
618 // replace proj by the nearest vertex?
619 std::vector<float> dists(vertexNumber_);
620 for(SimplexId i = 0; i < vertexNumber_; ++i) {
621 Point pv{};
622 triangulation.getVertexPoint(i, pv.x, pv.y, pv.z);
623 dists[i] = Geometry::distance(&pa.x, &pv.x);
624 }
625 auto min = std::min_element(dists.begin(), dists.end()) - dists.begin();
626 triangulation.getVertexPoint(min, res.x, res.y, res.z);
627 nearestVertex = min;
628 }
629
630 SimplexId projSucess = success ? (doReverseProj ? 1 : 2) : 3;
631
632 return std::make_tuple(res, nearestVertex, trChecked, projSucess);
633}

635template <typename triangulationType>
636int ttk::QuadrangulationSubdivision::project(
637 const std::set<size_t> &filtered,
638 const triangulationType &triangulation,
639 const bool lastIter) {
640 Timer tm;
641
642 if(lastIter) {
643 trianglesChecked_.clear();
644 projSucceeded_.clear();
645 trianglesChecked_.resize(outputPoints_.size());
646 projSucceeded_.resize(outputPoints_.size());
647 }
648
649 // temp storage for projected points
650 std::vector<Point> tmp(outputPoints_.size());
651
652 // main loop
653#ifdef TTK_ENABLE_OPENMP
654#pragma omp parallel for num_threads(threadNumber_)
655#endif // TTK_ENABLE_OPENMP
656 for(size_t i = 0; i < outputPoints_.size(); i++) {
657
658 // skip computation if i in filtered
659 if(filtered.find(i) != filtered.end()) {
660 tmp[i] = outputPoints_[i];
661 continue;
662 }
663
664 // replace curr in outputPoints_ by its projection
665 auto res = findProjection(i, ReverseProjection, triangulation);
666
667 tmp[i] = std::get<0>(res);
668 nearestVertexIdentifier_[i] = std::get<1>(res);
669
670 if(lastIter) {
671 // fill in debug info
672 trianglesChecked_[i] = std::get<2>(res);
673 projSucceeded_[i] = std::get<3>(res);
674 }
675 }
676
677 outputPoints_ = std::move(tmp);
678
679 this->printMsg("Projected "
680 + std::to_string(outputPoints_.size() - filtered.size())
681 + " points",
682 1.0, tm.getElapsedTime(), this->threadNumber_);
683
684 return 0;
685}

687template <typename triangulationType>
688int ttk::QuadrangulationSubdivision::subdivise(
689 const triangulationType &triangulation) {
690
691 using edgeType = std::pair<LongSimplexId, LongSimplexId>;
692 using vertexType = std::pair<LongSimplexId, Point>;
693 std::map<edgeType, vertexType> processedEdges;
694
695 // temp storage for quad subdivision
696 std::vector<Quad> tmp{};
697
698 Timer tm;
699
700 // avoid reallocation in loop, causing invalid pointers
701 outputPoints_.reserve(outputPoints_.size() * 5);
702
703 vertexDistance_.resize(outputPoints_.size());
704
705 // get all other vertices sharing a quad
706 quadNeighbors_.clear();
707 quadNeighbors_.resize(outputPoints_.size());
708 getQuadNeighbors(outputQuads_, quadNeighbors_, true);
709
710 // compute shortest distance from every vertex to all other that share a quad
711#ifdef TTK_ENABLE_OPENMP
712#pragma omp parallel for num_threads(threadNumber_)
713#endif // TTK_ENABLE_OPENMP
714 for(size_t i = 0; i < outputPoints_.size(); ++i) {
715
716 // skip if already computed on a coarser subdivision
717 if(vertexDistance_[i].empty()) {
718
719 // do not propagate on the whole mesh
720 std::vector<SimplexId> bounds;
721 for(auto &p : quadNeighbors_[i]) {
722 bounds.emplace_back(nearestVertexIdentifier_[p]);
723 }
724
725 Dijkstra::shortestPath(
726 nearestVertexIdentifier_[i], triangulation, vertexDistance_[i], bounds);
727 }
728 }
729
730 for(auto &q : outputQuads_) {
731
732 auto i = static_cast<size_t>(q[0]);
733 auto j = static_cast<size_t>(q[1]);
734 auto k = static_cast<size_t>(q[2]);
735 auto l = static_cast<size_t>(q[3]);
736
737 // middles of edges
738 auto ijid = findEdgeMiddle(i, j, triangulation);
739 auto jkid = findEdgeMiddle(j, k, triangulation);
740 auto klid = findEdgeMiddle(k, l, triangulation);
741 auto liid = findEdgeMiddle(l, i, triangulation);
742
743 Point midij{};
744 triangulation.getVertexPoint(ijid, midij.x, midij.y, midij.z);
745 Point midjk{};
746 triangulation.getVertexPoint(jkid, midjk.x, midjk.y, midjk.z);
747 Point midkl{};
748 triangulation.getVertexPoint(klid, midkl.x, midkl.y, midkl.z);
749 Point midli{};
750 triangulation.getVertexPoint(liid, midli.x, midli.y, midli.z);
751
752 std::vector<size_t> quadVertices{i, j, k, l};
753 // barycenter TTK identifier
754 auto baryid = findQuadBary(quadVertices);
755 // barycenter 3D coordinates
756 Point bary{};
757 triangulation.getVertexPoint(baryid, bary.x, bary.y, bary.z);
758
759 // order edges to avoid duplicates (ij vs. ji)
760 auto ij = std::make_pair(std::min(q[0], q[1]), std::max(q[0], q[1]));
761 auto jk = std::make_pair(std::min(q[1], q[2]), std::max(q[1], q[2]));
762 auto kl = std::make_pair(std::min(q[2], q[3]), std::max(q[2], q[3]));
763 auto li = std::make_pair(std::min(q[3], q[0]), std::max(q[3], q[0]));
764
765 auto process_edge_middle
766 = [&](const std::pair<LongSimplexId, LongSimplexId> &pair,
767 const Point &pt, const SimplexId id) {
768 /* check if edge already processed by a neighbor quad */
769 if(processedEdges.find(pair) == processedEdges.end()) {
770 processedEdges[pair] = std::make_pair(outputPoints_.size(), pt);
771 /* add new point 3d coordinates to vector of output points */
772 outputPoints_.emplace_back(pt);
773 /* new point is an edge middle */
774 outputVertType_.emplace_back(1);
775 /* store also TTK identifier of triangular mesh vertex */
776 nearestVertexIdentifier_.emplace_back(id);
777 }
778 };
779
780 process_edge_middle(ij, midij, ijid);
781 process_edge_middle(jk, midjk, jkid);
782 process_edge_middle(kl, midkl, klid);
783 process_edge_middle(li, midli, liid);
784
785 // barycenter index in outputPoints_
786 auto baryIdx = static_cast<LongSimplexId>(outputPoints_.size());
787 outputPoints_.emplace_back(bary);
788 outputVertType_.emplace_back(2);
789 nearestVertexIdentifier_.emplace_back(baryid);
790
791 // add the four new quads
792 tmp.emplace_back(
793 Quad{q[0], processedEdges[ij].first, baryIdx, processedEdges[li].first});
794 tmp.emplace_back(
795 Quad{q[1], processedEdges[jk].first, baryIdx, processedEdges[ij].first});
796 tmp.emplace_back(
797 Quad{q[2], processedEdges[kl].first, baryIdx, processedEdges[jk].first});
798 tmp.emplace_back(
799 Quad{q[3], processedEdges[li].first, baryIdx, processedEdges[kl].first});
800 }
801
802 // output subdivision level
803 auto currSubd = outputSubdivision_.back() + 1;
804 auto subdBeg = outputSubdivision_.size();
805 outputSubdivision_.resize(outputPoints_.size());
806 std::fill(
807 outputSubdivision_.begin() + subdBeg, outputSubdivision_.end(), currSubd);
808
809 this->printMsg("Subdivised " + std::to_string(outputQuads_.size())
810 + " quads into " + std::to_string(tmp.size())
811 + " new quads (" + std::to_string(outputPoints_.size())
812 + " points)",
813 1.0, tm.getElapsedTime(), this->threadNumber_);
814
815 outputQuads_ = std::move(tmp);
816
817 return 0;
818}

820template <typename triangulationType>
821void ttk::QuadrangulationSubdivision::quadStatistics(
822 const triangulationType &triangulation) {
823 Timer tm;
824
825 quadArea_.clear();
826 quadArea_.resize(outputQuads_.size());
827 quadDiagsRatio_.clear();
828 quadDiagsRatio_.resize(outputQuads_.size());
829 quadEdgesRatio_.clear();
830 quadEdgesRatio_.resize(outputQuads_.size());
831 quadAnglesRatio_.clear();
832 quadAnglesRatio_.resize(outputQuads_.size());
833 hausdorff_.clear();
834 hausdorff_.resize(outputPoints_.size());
835
836#ifdef TTK_ENABLE_OPENMP
837#pragma omp parallel for num_threads(threadNumber_)
838#endif // TTK_ENABLE_OPENMP
839 for(size_t i = 0; i < outputQuads_.size(); ++i) {
840 const auto &q = outputQuads_[i];
841 Point pi = outputPoints_[q[0]];
842 Point pj = outputPoints_[q[1]];
843 Point pk = outputPoints_[q[2]];
844 Point pl = outputPoints_[q[3]];
845
846 // quadrangle area
847 float area0{}, area1{};
848 Geometry::computeTriangleArea(&pi.x, &pj.x, &pk.x, area0);
849 Geometry::computeTriangleArea(&pi.x, &pk.x, &pl.x, area1);
850 quadArea_[i] = area0 + area1;
851
852 // diagonals ratio
853 auto diag0 = Geometry::distance(&pi.x, &pk.x);
854 auto diag1 = Geometry::distance(&pj.x, &pl.x);
855 quadDiagsRatio_[i] = std::min(diag0, diag1) / std::max(diag0, diag1);
856
857 // edges ratio
858 std::array<float, 4> edges{
859 Geometry::distance(&pi.x, &pj.x), // ij
860 Geometry::distance(&pj.x, &pk.x), // jk
861 Geometry::distance(&pk.x, &pl.x), // kl
862 Geometry::distance(&pl.x, &pi.x), // li
863 };
864 quadEdgesRatio_[i] = *std::min_element(edges.begin(), edges.end())
865 / *std::max_element(edges.begin(), edges.end());
866
867 // angles ratio
868 std::array<float, 4> angles{
869 Geometry::angle(&pi.x, &pl.x, &pi.x, &pj.x), // lij
870 Geometry::angle(&pj.x, &pi.x, &pj.x, &pk.x), // ijk
871 Geometry::angle(&pk.x, &pj.x, &pk.x, &pl.x), // jkl
872 Geometry::angle(&pl.x, &pk.x, &pl.x, &pi.x), // kli
873 };
874 quadAnglesRatio_[i] = *std::min_element(angles.begin(), angles.end())
875 / *std::max_element(angles.begin(), angles.end());
876 }
877
878 // compute ratio between quad area and mean quad area
879
880 // global surface area
881 float sumArea{};
882 for(const auto a : quadArea_) {
883 sumArea += a;
884 }
885 for(auto &a : quadArea_) {
886 a *= quadArea_.size() / sumArea;
887 }
888
889 // compute the minimal distance from every triangulation point to
890 // every quadrangulation point
891
892 std::vector<size_t> triVertsDist(vertexNumber_);
893
894#ifdef TTK_ENABLE_OPENMP
895#pragma omp parallel for num_threads(threadNumber_)
896#endif // TTK_ENABLE_OPENMP
897 for(size_t i = 0; i < static_cast<size_t>(vertexNumber_); ++i) {
898 float minDist{std::numeric_limits<float>::infinity()};
899 Point p{};
900 triangulation.getVertexPoint(i, p.x, p.y, p.z);
901
902 for(size_t j = 0; j < outputPoints_.size(); ++j) {
903 auto dist = Geometry::distance(&p.x, &outputPoints_[j].x);
904 if(dist < minDist) {
905 minDist = dist;
906 triVertsDist[i] = j;
907 }
908 }
909 }
910
911 // compute triangulation bounding box diagonal
912 Point pmin{std::numeric_limits<float>::infinity(),
913 std::numeric_limits<float>::infinity(),
914 std::numeric_limits<float>::infinity()};
915 Point pmax{-std::numeric_limits<float>::infinity(),
916 -std::numeric_limits<float>::infinity(),
917 -std::numeric_limits<float>::infinity()};
918
919 for(size_t i = 0; i < static_cast<size_t>(vertexNumber_); ++i) {
920 Point p{};
921 triangulation.getVertexPoint(i, p.x, p.y, p.z);
922 pmax.x = std::max(pmax.x, p.x);
923 pmax.y = std::max(pmax.y, p.y);
924 pmax.z = std::max(pmax.z, p.z);
925 pmin.x = std::min(pmin.x, p.x);
926 pmin.y = std::min(pmin.y, p.y);
927 pmin.z = std::min(pmin.z, p.z);
928 }
929
930 auto bboxDiag = Geometry::distance(&pmin.x, &pmax.x);
931
932#ifdef TTK_ENABLE_OPENMP
933#pragma omp parallel for num_threads(threadNumber_)
934#endif // TTK_ENABLE_OPENMP
935 for(size_t i = 0; i < outputPoints_.size(); ++i) {
936 float maxDist{};
937 for(size_t j = 0; j < static_cast<size_t>(vertexNumber_); ++j) {
938 Point p{};
939 triangulation.getVertexPoint(j, p.x, p.y, p.z);
940
941 if(triVertsDist[j] == i) {
942 auto dist = Geometry::distance(&p.x, &outputPoints_[i].x);
943 if(dist > maxDist) {
944 maxDist = dist;
945 }
946 }
947 }
948 hausdorff_[i] = maxDist / bboxDiag / vertexNumber_ * 1e8;
949 }
950
951 this->printMsg("Computed quad statistics", 1.0, tm.getElapsedTime(),
952 debug::LineMode::NEW, debug::Priority::DETAIL);
953}

955// main routine
956template <typename triangulationType>
957int ttk::QuadrangulationSubdivision::execute(
958 const triangulationType &triangulation) {
959
960 this->printMsg("Beginning computation...");
961 Timer tm;
962
963 // clear output variables
964 clearData();
965
966 // ensure consistency of dependent options
967 if(LockAllInputVertices) {
968 LockInputExtrema = true;
969 }
970
971 // store input points (MSC critical points)
972 for(size_t i = 0; i < inputVertexNumber_; i++) {
973 outputPoints_.emplace_back(inputVertices_[i]);
974 }
975
976 // copy input quads into vector
977 for(size_t i = 0; i < inputQuadNumber_; i++) {
978 outputQuads_.emplace_back(inputQuads_[i]);
979 }
980
981 // fill outputInfos_ with input data (critical points)
982 outputVertType_.resize(outputPoints_.size());
983 std::fill(outputVertType_.begin(), outputVertType_.end(), 0);
984
985 // fill outputSubdivision with input data
986 outputSubdivision_.resize(outputPoints_.size());
987 std::fill(outputSubdivision_.begin(), outputSubdivision_.end(), 0);
988
989 // vertices to filter from relaxation, projection
990 std::set<size_t> filtered{};
991 if(!LockAllInputVertices) {
992 if(LockInputExtrema) {
993 // get extraordinary vertices
994 findExtraordinaryVertices(filtered);
995 }
996 } else {
997 // fill vector with all input points indices from 0 to inputVertexNumber_
998 for(size_t i = 0; i < inputVertexNumber_; ++i) {
999 filtered.insert(i);
1000 }
1001 }
1002
1003 // main loop
1004 for(size_t i = 0; i < SubdivisionLevel; i++) {
1005 // subdivise each quadrangle by creating five new points, at the
1006 // center of each edge (4) and at the barycenter of the four
1007 // vertices (1).
1008 subdivise(triangulation);
1009 }
1010
1011 // retrieve mapping between every vertex and its neighbors
1012 quadNeighbors_.clear();
1013 quadNeighbors_.resize(outputPoints_.size());
1014 getQuadNeighbors(outputQuads_, quadNeighbors_);
1015
1016 // "relax" the new points, i.e. replace it by the barycenter of its
1017 // four neighbors
1018 for(size_t i = 0; i < RelaxationIterations; i++) {
1019 relax(filtered);
1020
1021 // project all points on the nearest triangle (except MSC critical
1022 // points)
1023 project(filtered, triangulation, (i == RelaxationIterations - 1));
1024 }
1025
1026 // compute valence of every quadrangle vertex
1027 outputValences_.resize(outputPoints_.size());
1028 std::transform(
1029 quadNeighbors_.begin(), quadNeighbors_.end(), outputValences_.begin(),
1030 [&](const std::set<size_t> &neighbors) { return neighbors.size(); });
1031
1032 quadStatistics(triangulation);
1033
1034 bool criterion = false;
1035 for(size_t i = 0; i < outputPoints_.size(); ++i) {
1036 if(outputValences_[i] > 4) {
1037 continue;
1038 }
1039 if(hausdorff_[i] > HausdorffLevel) {
1040 criterion = true;
1041 break;
1042 }
1043 }
1044
1045 if(criterion) {
1046 // log, clean & early return
1047 this->printErr("The output quadrangulation exceeds the provided Haussdorff "
1048 "distance tolerance");
1049 if(!ShowResError) {
1050 clearData();
1051 return 1;
1052 }
1053 }
1054
1055 this->printMsg("Produced " + std::to_string(outputQuads_.size())
1056 + " quadrangles with " + std::to_string(outputPoints_.size())
1057 + " points",
1058 1.0, tm.getElapsedTime(), this->threadNumber_);
1059
1060 return 0;
1061}
