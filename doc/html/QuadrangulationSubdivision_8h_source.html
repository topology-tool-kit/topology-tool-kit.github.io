342 template <typename triangulationType>
343 ttk::SimplexId ttk::QuadrangulationSubdivision::findEdgeMiddle(
344  const size_t a,
345  const size_t b,
346  const triangulationType &triangulation) const {
347 
348  std::vector<SimplexId> midId(this->threadNumber_);
349  std::vector<float> minValue(
350  this->threadNumber_, std::numeric_limits<float>::infinity());
351 
352  // euclidian barycenter of a and b
353  Point edgeEuclBary = (outputPoints_[a] + outputPoints_[b]) * 0.5F;
354 
355 #ifdef TTK_ENABLE_OPENMP
356 #pragma omp parallel for num_threads(threadNumber_)
357 #endif // TTK_ENABLE_OPENMP
358  for(size_t i = 0; i < vertexDistance_[a].size(); ++i) {
359 #ifdef TTK_ENABLE_OPENMP
360  const auto tid = omp_get_thread_num();
361 #else
362  const auto tid = 0;
363 #endif // TTK_ENABLE_OPENMP
364  float m = vertexDistance_[a][i];
365  float n = vertexDistance_[b][i];
366  // stay on the shortest path between a and b
367  float sum = m + n;
368 
369  // skip further computation
370  if(sum > minValue[tid]) {
371  continue;
372  }
373 
374  if(m != std::numeric_limits<float>::infinity()
375  && n != std::numeric_limits<float>::infinity()) {
376  // try to get the middle of the shortest path
377  sum += std::abs(m - n);
378  }
379 
380  // get the euclidian distance to AB
381  Point curr{};
382  triangulation.getVertexPoint(i, curr.x, curr.y, curr.z);
383  // try to minimize the euclidian distance to AB too
384  sum += Geometry::distance(&curr.x, &edgeEuclBary.x);
385 
386  // search for the minimizing index
387  if(sum < minValue[tid]) {
388  minValue[tid] = sum;
389  midId[tid] = i;
390  }
391  }
392 
393 #ifdef TTK_ENABLE_OPENMP
394  for(int i = 1; i < this->threadNumber_; ++i) {
395  if(minValue[i] < minValue[0]) {
396  minValue[0] = minValue[i];
397  midId[0] = midId[i];
398  }
399  }
400 #endif // TTK_ENABLE_OPENMP
401 
402  return midId[0];
403 }

409  ttk::SimplexId>
410  ttk::QuadrangulationSubdivision::findProjection(
411  const size_t a,
412  const bool forceReverseProj,
413  const triangulationType &triangulation) const {
414 
415  static const float PREC_FLT{powf(10, -FLT_DIG)};
416 
417  // current vertex 3d coordinates
418  Point pa = outputPoints_[a];
419 
420  Point res{};
421 
422  // fallback to euclidian projection code if no normals
423  bool doReverseProj = forceReverseProj;
424 
425  // quad normal in a
426  Point normalsMean{};
427 
428  if(doReverseProj) {
429  // compute mean of normals
430  normalsMean = getQuadNormal(a);
431 
432  doReverseProj = !std::isnan(normalsMean.x);
433  }
434 
435  // found a projection in one triangle
436  bool success = false;
437  // list of triangle IDs to test to find a potential projection
438  std::stack<SimplexId> trianglesToTest;
439  // list of triangle IDs already tested
440  // (takes more memory to reduce computation time)
441  std::vector<bool> trianglesTested(
442  triangulation.getNumberOfTriangles(), false);
443  // number of triangles tested
444  size_t trChecked{0};
445  // vertex in triangle with highest barycentric coordinate
446  SimplexId nearestVertex = nearestVertexIdentifier_[a];
447 
448  // number of triangles around nearest vertex
449  SimplexId triangleNumber
450  = triangulation.getVertexTriangleNumber(nearestVertex);
451  // init pipeline by checking in every triangle around selected vertex
452  for(SimplexId j = 0; j < triangleNumber; j++) {
453  SimplexId ntid;
454  triangulation.getVertexTriangle(nearestVertex, j, ntid);
455  trianglesToTest.push(ntid);
456  }
457 
458  while(!trianglesToTest.empty()) {
459  SimplexId i = trianglesToTest.top();
460  trianglesToTest.pop();
461 
462  // skip if already tested
463  if(trianglesTested[i]) {
464  continue;
465  }
466 
467  // get triangle vertices
468  std::array<SimplexId, 3> tverts{};
469  triangulation.getTriangleVertex(i, 0, tverts[0]);
470  triangulation.getTriangleVertex(i, 1, tverts[1]);
471  triangulation.getTriangleVertex(i, 2, tverts[2]);
472 
473  // get coordinates of triangle vertices
474  Point pm{}, pn{}, po{};
475  triangulation.getVertexPoint(tverts[0], pm.x, pm.y, pm.z);
476  triangulation.getVertexPoint(tverts[1], pn.x, pn.y, pn.z);
477  triangulation.getVertexPoint(tverts[2], po.x, po.y, po.z);
478 
479  // triangle normal: cross product of two edges
480  Point crossP{};
481  // mn, mo vectors
482  Point mn = pn - pm;
483  Point mo = po - pm;
484  // compute mn ^ mo
485  Geometry::crossProduct(&mn.x, &mo.x, &crossP.x);
486  // unitary normal vector
487  Point normTri = crossP / Geometry::magnitude(&crossP.x);
488 
489  // compute intersection of triangle plane and line (a, normalsMean)
490  if(doReverseProj) {
491 
492  auto denom = Geometry::dotProduct(&normalsMean.x, &normTri.x);
493 
494  // check if triangle plane is parallel to quad normal
495  if(std::abs(denom) < PREC_FLT) {
496  // skip this iteration after filling pipeline
497  trianglesTested[i] = true;
498  // fill pipeline with neighboring triangles
499  for(auto &vert : tverts) {
500  auto ntr = triangulation.getVertexTriangleNumber(vert);
501  for(SimplexId j = 0; j < ntr; ++j) {
502  SimplexId tid;
503  triangulation.getVertexTriangle(vert, j, tid);
504  if(tid != i) {
505  trianglesToTest.push(tid);
506  }
507  }
508  }
509  continue;
510  }
511 
512  // use formula from Wikipedia: line-plane intersection
513  auto tmp = pm - pa;
514  auto alpha = Geometry::dotProduct(&tmp.x, &normTri.x) / denom;
515 
516  // intersection
517  res = pa + normalsMean * alpha;
518 
519  }
520  // compute euclidian projection of a in triangle plane
521  else {
522 
523  auto tmp = pa - pm;
524  // projection
525  res = pa - normTri * Geometry::dotProduct(&normTri.x, &tmp.x);
526  }
527 
528  // compute barycentric coords of projection
529  std::vector<float> baryCoords;
530  Geometry::computeBarycentricCoordinates(
531  &pm.x, &pn.x, &po.x, &res.x, baryCoords);
532 
533  // check if projection in triangle
534  bool inTriangle = true;
535  for(auto &coord : baryCoords) {
536  if(coord < PREC_FLT) {
537  inTriangle = false;
538  }
539  if(coord > 1 + PREC_FLT) {
540  inTriangle = false;
541  }
542  }
543 
544  // mark triangle as tested
545  trianglesTested[i] = true;
546  trChecked++;
547 
548  if(inTriangle) {
549  success = true;
550  // should we check if we have the nearest triangle?
551  break;
552  }
553 
554  // extrema values in baryCoords
555  auto extrema = std::minmax_element(baryCoords.begin(), baryCoords.end());
556 
557  // find the nearest triangle vertices (with the highest/positive
558  // values in baryCoords) from proj
559  std::vector<SimplexId> vertices(2);
560  vertices[0] = tverts[extrema.second - baryCoords.begin()];
561  for(size_t j = 0; j < baryCoords.size(); j++) {
562  if(j != static_cast<size_t>(extrema.first - baryCoords.begin())
563  && j != static_cast<size_t>(extrema.second - baryCoords.begin())) {
564  vertices[1] = tverts[j];
565  break;
566  }
567  }
568 
569  // store vertex with highest barycentric coordinate
570  nearestVertex = vertices[0];
571 
572  // triangles around vertices[0] and vertices[1]
573  std::array<std::set<SimplexId>, 2> vertsTriangles{};
574 
575  // get triangles around vertices
576  for(size_t j = 0; j < vertices.size(); ++j) {
577  SimplexId tnum = triangulation.getVertexTriangleNumber(vertices[j]);
578  for(SimplexId k = 0; k < tnum; k++) {
579  SimplexId tid;
580  triangulation.getVertexTriangle(vertices[j], k, tid);
581  if(tid == i) {
582  continue;
583  }
584  vertsTriangles[j].insert(tid);
585  }
586  }
587 
588  // triangles to test next
589  std::vector<SimplexId> common_triangles;
590 
591  // look for triangles sharing the vertices with max values in baryCoords
592  std::set_intersection(vertsTriangles[0].begin(), vertsTriangles[0].end(),
593  vertsTriangles[1].begin(), vertsTriangles[1].end(),
594  std::back_inserter(common_triangles));
595 
596  for(auto &ntid : common_triangles) {
597  if(!trianglesTested[ntid]) {
598  trianglesToTest.push(ntid);
599  }
600  }
601  }
602 
603  const size_t maxTrChecked = 100;
604 
605  if(success && trChecked > maxTrChecked) {
606  success = false;
607  }
608 
609  if(!success) {
610  if(!forceReverseProj) {
611  return findProjection(a, true, triangulation);
612  }
613  // replace proj by the nearest vertex?
614  std::vector<float> dists(vertexNumber_);
615  for(SimplexId i = 0; i < vertexNumber_; ++i) {
616  Point pv{};
617  triangulation.getVertexPoint(i, pv.x, pv.y, pv.z);
618  dists[i] = Geometry::distance(&pa.x, &pv.x);
619  }
620  auto min = std::min_element(dists.begin(), dists.end()) - dists.begin();
621  triangulation.getVertexPoint(min, res.x, res.y, res.z);
622  nearestVertex = min;
623  }
624 
625  SimplexId projSucess = success ? (doReverseProj ? 1 : 2) : 3;
626 
627  return std::make_tuple(res, nearestVertex, trChecked, projSucess);
628 }

682 template <typename triangulationType>
683 int ttk::QuadrangulationSubdivision::subdivise(
684  const triangulationType &triangulation) {
685 
686  using edgeType = std::pair<LongSimplexId, LongSimplexId>;
687  using vertexType = std::pair<LongSimplexId, Point>;
688  std::map<edgeType, vertexType> processedEdges;
689 
690  // temp storage for quad subdivision
691  std::vector<Quad> tmp{};
692 
693  Timer tm;
694 
695  // avoid reallocation in loop, causing invalid pointers
696  outputPoints_.reserve(outputPoints_.size() * 5);
697 
698  vertexDistance_.resize(outputPoints_.size());
699 
700  // get all other vertices sharing a quad
701  quadNeighbors_.clear();
702  quadNeighbors_.resize(outputPoints_.size());
703  getQuadNeighbors(outputQuads_, quadNeighbors_, true);
704 
705  // compute shortest distance from every vertex to all other that share a quad
706 #ifdef TTK_ENABLE_OPENMP
707 #pragma omp parallel for num_threads(threadNumber_)
708 #endif // TTK_ENABLE_OPENMP
709  for(size_t i = 0; i < outputPoints_.size(); ++i) {
710 
711  // skip if already computed on a coarser subdivision
712  if(vertexDistance_[i].empty()) {
713 
714  // do not propagate on the whole mesh
715  std::vector<SimplexId> bounds;
716  for(auto &p : quadNeighbors_[i]) {
717  bounds.emplace_back(nearestVertexIdentifier_[p]);
718  }
719 
720  Dijkstra::shortestPath(
721  nearestVertexIdentifier_[i], triangulation, vertexDistance_[i], bounds);
722  }
723  }
724 
725  for(auto &q : outputQuads_) {
726 
727  auto i = static_cast<size_t>(q[0]);
728  auto j = static_cast<size_t>(q[1]);
729  auto k = static_cast<size_t>(q[2]);
730  auto l = static_cast<size_t>(q[3]);
731 
732  // middles of edges
733  auto ijid = findEdgeMiddle(i, j, triangulation);
734  auto jkid = findEdgeMiddle(j, k, triangulation);
735  auto klid = findEdgeMiddle(k, l, triangulation);
736  auto liid = findEdgeMiddle(l, i, triangulation);
737 
738  Point midij{};
739  triangulation.getVertexPoint(ijid, midij.x, midij.y, midij.z);
740  Point midjk{};
741  triangulation.getVertexPoint(jkid, midjk.x, midjk.y, midjk.z);
742  Point midkl{};
743  triangulation.getVertexPoint(klid, midkl.x, midkl.y, midkl.z);
744  Point midli{};
745  triangulation.getVertexPoint(liid, midli.x, midli.y, midli.z);
746 
747  std::vector<size_t> quadVertices{i, j, k, l};
748  // barycenter TTK identifier
749  auto baryid = findQuadBary(quadVertices);
750  // barycenter 3D coordinates
751  Point bary{};
752  triangulation.getVertexPoint(baryid, bary.x, bary.y, bary.z);
753 
754  // order edges to avoid duplicates (ij vs. ji)
755  auto ij = std::make_pair(std::min(q[0], q[1]), std::max(q[0], q[1]));
756  auto jk = std::make_pair(std::min(q[1], q[2]), std::max(q[1], q[2]));
757  auto kl = std::make_pair(std::min(q[2], q[3]), std::max(q[2], q[3]));
758  auto li = std::make_pair(std::min(q[3], q[0]), std::max(q[3], q[0]));
759 
760  auto process_edge_middle
761  = [&](const std::pair<LongSimplexId, LongSimplexId> &pair,
762  const Point &pt, const SimplexId id) {
763  /* check if edge already processed by a neighbor quad */
764  if(processedEdges.find(pair) == processedEdges.end()) {
765  processedEdges[pair] = std::make_pair(outputPoints_.size(), pt);
766  /* add new point 3d coordinates to vector of output points */
767  outputPoints_.emplace_back(pt);
768  /* new point is an edge middle */
769  outputVertType_.emplace_back(1);
770  /* store also TTK identifier of triangular mesh vertex */
771  nearestVertexIdentifier_.emplace_back(id);
772  }
773  };
774 
775  process_edge_middle(ij, midij, ijid);
776  process_edge_middle(jk, midjk, jkid);
777  process_edge_middle(kl, midkl, klid);
778  process_edge_middle(li, midli, liid);
779 
780  // barycenter index in outputPoints_
781  auto baryIdx = static_cast<LongSimplexId>(outputPoints_.size());
782  outputPoints_.emplace_back(bary);
783  outputVertType_.emplace_back(2);
784  nearestVertexIdentifier_.emplace_back(baryid);
785 
786  // add the four new quads
787  tmp.emplace_back(
788  Quad{q[0], processedEdges[ij].first, baryIdx, processedEdges[li].first});
789  tmp.emplace_back(
790  Quad{q[1], processedEdges[jk].first, baryIdx, processedEdges[ij].first});
791  tmp.emplace_back(
792  Quad{q[2], processedEdges[kl].first, baryIdx, processedEdges[jk].first});
793  tmp.emplace_back(
794  Quad{q[3], processedEdges[li].first, baryIdx, processedEdges[kl].first});
795  }
796 
797  // output subdivision level
798  auto currSubd = outputSubdivision_.back() + 1;
799  auto subdBeg = outputSubdivision_.size();

815 template <typename triangulationType>
816 void ttk::QuadrangulationSubdivision::quadStatistics(
817  const triangulationType &triangulation) {
818  Timer tm;
819 
820  quadArea_.clear();
821  quadArea_.resize(outputQuads_.size());
822  quadDiagsRatio_.clear();
823  quadDiagsRatio_.resize(outputQuads_.size());
824  quadEdgesRatio_.clear();
825  quadEdgesRatio_.resize(outputQuads_.size());
826  quadAnglesRatio_.clear();
827  quadAnglesRatio_.resize(outputQuads_.size());
828  hausdorff_.clear();
829  hausdorff_.resize(outputPoints_.size());
830 
831 #ifdef TTK_ENABLE_OPENMP
832 #pragma omp parallel for num_threads(threadNumber_)
833 #endif // TTK_ENABLE_OPENMP
834  for(size_t i = 0; i < outputQuads_.size(); ++i) {
835  const auto &q = outputQuads_[i];
836  Point pi = outputPoints_[q[0]];
837  Point pj = outputPoints_[q[1]];
838  Point pk = outputPoints_[q[2]];
839  Point pl = outputPoints_[q[3]];
840 
841  // quadrangle area
842  float area0{}, area1{};
843  Geometry::computeTriangleArea(&pi.x, &pj.x, &pk.x, area0);
844  Geometry::computeTriangleArea(&pi.x, &pk.x, &pl.x, area1);
845  quadArea_[i] = area0 + area1;
846 
847  // diagonals ratio
848  auto diag0 = Geometry::distance(&pi.x, &pk.x);
849  auto diag1 = Geometry::distance(&pj.x, &pl.x);
850  quadDiagsRatio_[i] = std::min(diag0, diag1) / std::max(diag0, diag1);
851 
852  // edges ratio
853  std::array<float, 4> edges{
854  Geometry::distance(&pi.x, &pj.x), // ij
855  Geometry::distance(&pj.x, &pk.x), // jk
856  Geometry::distance(&pk.x, &pl.x), // kl
857  Geometry::distance(&pl.x, &pi.x), // li
858  };
859  quadEdgesRatio_[i] = *std::min_element(edges.begin(), edges.end())
860  / *std::max_element(edges.begin(), edges.end());
861 
862  // angles ratio
863  std::array<float, 4> angles{
864  Geometry::angle(&pi.x, &pl.x, &pi.x, &pj.x), // lij
865  Geometry::angle(&pj.x, &pi.x, &pj.x, &pk.x), // ijk
866  Geometry::angle(&pk.x, &pj.x, &pk.x, &pl.x), // jkl
867  Geometry::angle(&pl.x, &pk.x, &pl.x, &pi.x), // kli
868  };
869  quadAnglesRatio_[i] = *std::min_element(angles.begin(), angles.end())
870  / *std::max_element(angles.begin(), angles.end());
871  }
872 
873  // compute ratio between quad area and mean quad area
874 
875  // global surface area
876  float sumArea{};
877  for(const auto a : quadArea_) {
878  sumArea += a;
879  }
880  for(auto &a : quadArea_) {
881  a *= quadArea_.size() / sumArea;
882  }
883 
884  // compute the minimal distance from every triangulation point to
885  // every quadrangulation point
886 
887  std::vector<size_t> triVertsDist(vertexNumber_);
888 
889 #ifdef TTK_ENABLE_OPENMP
890 #pragma omp parallel for num_threads(threadNumber_)
891 #endif // TTK_ENABLE_OPENMP
892  for(size_t i = 0; i < static_cast<size_t>(vertexNumber_); ++i) {
893  float minDist{std::numeric_limits<float>::infinity()};
894  Point p{};
895  triangulation.getVertexPoint(i, p.x, p.y, p.z);
896 
897  for(size_t j = 0; j < outputPoints_.size(); ++j) {
898  auto dist = Geometry::distance(&p.x, &outputPoints_[j].x);
899  if(dist < minDist) {
900  minDist = dist;
901  triVertsDist[i] = j;
902  }
903  }
904  }
905 
906  // compute triangulation bounding box diagonal
907  Point pmin{std::numeric_limits<float>::infinity(),
908  std::numeric_limits<float>::infinity(),
909  std::numeric_limits<float>::infinity()};
910  Point pmax{-std::numeric_limits<float>::infinity(),
911  -std::numeric_limits<float>::infinity(),
912  -std::numeric_limits<float>::infinity()};
913 
914  for(size_t i = 0; i < static_cast<size_t>(vertexNumber_); ++i) {
915  Point p{};
916  triangulation.getVertexPoint(i, p.x, p.y, p.z);
917  pmax.x = std::max(pmax.x, p.x);
918  pmax.y = std::max(pmax.y, p.y);
919  pmax.z = std::max(pmax.z, p.z);
920  pmin.x = std::min(pmin.x, p.x);
921  pmin.y = std::min(pmin.y, p.y);
922  pmin.z = std::min(pmin.z, p.z);
923  }
924 
925  auto bboxDiag = Geometry::distance(&pmin.x, &pmax.x);
926 
927 #ifdef TTK_ENABLE_OPENMP
928 #pragma omp parallel for num_threads(threadNumber_)
929 #endif // TTK_ENABLE_OPENMP
930  for(size_t i = 0; i < outputPoints_.size(); ++i) {
931  float maxDist{};
932  for(size_t j = 0; j < static_cast<size_t>(vertexNumber_); ++j) {
933  Point p{};
934  triangulation.getVertexPoint(j, p.x, p.y, p.z);
935 
936  if(triVertsDist[j] == i) {
937  auto dist = Geometry::distance(&p.x, &outputPoints_[i].x);
938  if(dist > maxDist) {
939  maxDist = dist;
940  }
941  }
942  }
943  hausdorff_[i] = maxDist / bboxDiag / vertexNumber_ * 1e8;
944  }
945 
946  this->printMsg("Computed quad statistics", 1.0, tm.getElapsedTime(),
947  debug::LineMode::NEW, debug::Priority::DETAIL);
948 }

951 template <typename triangulationType>
952 int ttk::QuadrangulationSubdivision::execute(
953  const triangulationType &triangulation) {
954 
955  this->printMsg("Beginning computation...");
956  Timer tm;
957 
958  // clear output variables
959  clearData();
960 
961  // ensure consistency of dependent options
962  if(LockAllInputVertices) {
963  LockInputExtrema = true;
964  }
965 
966  // store input points (MSC critical points)
967  for(size_t i = 0; i < inputVertexNumber_; i++) {
968  outputPoints_.emplace_back(inputVertices_[i]);
969  }
970 
971  // copy input quads into vector
972  for(size_t i = 0; i < inputQuadNumber_; i++) {
973  outputQuads_.emplace_back(inputQuads_[i]);
974  }
975 
976  // fill outputInfos_ with input data (critical points)
977  outputVertType_.resize(outputPoints_.size());
978  std::fill(outputVertType_.begin(), outputVertType_.end(), 0);
979 
980  // fill outputSubdivision with input data
981  outputSubdivision_.resize(outputPoints_.size());
982  std::fill(outputSubdivision_.begin(), outputSubdivision_.end(), 0);
983 
984  // vertices to filter from relaxation, projection
985  std::set<size_t> filtered{};
986  if(!LockAllInputVertices) {
987  if(LockInputExtrema) {
988  // get extraordinary vertices
989  findExtraordinaryVertices(filtered);
990  }
991  } else {
992  // fill vector with all input points indices from 0 to inputVertexNumber_
993  for(size_t i = 0; i < inputVertexNumber_; ++i) {
994  filtered.insert(i);
995  }
996  }
997 
998  // main loop
999  for(size_t i = 0; i < SubdivisionLevel; i++) {
1000  // subdivise each quadrangle by creating five new points, at the
1001  // center of each edge (4) and at the barycenter of the four
1002  // vertices (1).
1003  subdivise(triangulation);
1004  }
1005 
1006  // retrieve mapping between every vertex and its neighbors
1007  quadNeighbors_.clear();
1008  quadNeighbors_.resize(outputPoints_.size());
1009  getQuadNeighbors(outputQuads_, quadNeighbors_);
1010 
1011  // "relax" the new points, i.e. replace it by the barycenter of its
1012  // four neighbors
1013  for(size_t i = 0; i < RelaxationIterations; i++) {
1014  relax(filtered);
1015 
1016  // project all points on the nearest triangle (except MSC critical
1017  // points)
1018  project(filtered, triangulation, (i == RelaxationIterations - 1));
1019  }
1020 
1021  // compute valence of every quadrangle vertex
1022  outputValences_.resize(outputPoints_.size());
1023  std::transform(
1024  quadNeighbors_.begin(), quadNeighbors_.end(), outputValences_.begin(),
1025  [&](const std::set<size_t> &neighbors) { return neighbors.size(); });
1026 
1027  quadStatistics(triangulation);
1028 
1029  bool criterion = false;
1030  for(size_t i = 0; i < outputPoints_.size(); ++i) {
1031  if(outputValences_[i] > 4) {
1032  continue;
1033  }
1034  if(hausdorff_[i] > HausdorffLevel) {
1035  criterion = true;
1036  break;
1037  }
1038  }
1039 
1040  if(criterion) {
1041  // log, clean & early return
1042  this->printErr("The output quadrangulation exceeds the provided Haussdorff "
1043  "distance tolerance");
1044  if(!ShowResError) {
1045  clearData();
1046  return 1;
1047  }
1048  }
1049 
1050  this->printMsg("Produced " + std::to_string(outputQuads_.size())
1051  + " quadrangles with " + std::to_string(outputPoints_.size())
1052  + " points",
1053  1.0, tm.getElapsedTime(), this->threadNumber_);
1054 
1055  return 0;
1056 }
