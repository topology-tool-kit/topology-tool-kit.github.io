308  // This implementation is based on a naive nearest neighbor search
309  SimplexId minv = 0;
310  float mind = compDist2(minv, p);
311  const auto nv

313  for(SimplexId v = 1; v < nv; ++v) {

324 template <typename scalarT, class Triang>
325 void ttk::ContourAroundPoint::handleOneInpPt(SimplexId vBeg,
326  float isoval,
327  int flag,
328  float scalar) const {
329  auto triang = reinterpret_cast<Triang *>(_inpFldTriang);
330  auto inpScalars = reinterpret_cast<const scalarT *>(_inpFldScalars);
331  const bool vBegIsMin = flag == 0;
332 
333  // The general idea of the algorithm is to start a (breadth-first) search
334  // on the triangulation starting at the input vertex.
335  // Until the isocontour is met, all vertices are collected to approximate
336  // the (weighted) centroid of the enclosed area/volume.
337  // The edges leading outside of the region are also collected, to compute
338  // the actual isocontour (that intersects these edges).
339 
340  auto innerVerts = std::vector<SimplexId>{vBeg};
341  std::set<SimplexId> xEdges; // set data-structure is needed in addOutput
342 
343  struct VertEdge {
344  SimplexId v; // where am i
345  SimplexId e; // how did i get here
346  };
347  std::vector<VertEdge> q;
348  std::set<SimplexId> handledEdges;
349 
350  auto enqueueNeighbors = [triang, &q, &handledEdges](SimplexId vSrc) {
351  const auto n = triang->getVertexEdgeNumber(vSrc);
352  for(SimplexId i = 0; i < n; ++i) {
353  SimplexId e;
354  triang->getVertexEdge(vSrc, i, e);
355  if(handledEdges.count(e))
356  continue;
357  handledEdges.insert(e);
358  // We assume the local indexing of vertex neighbors is the same
359  // as for the edges (edge 0 leads to neighbor 0, etc.).
360  SimplexId vTgt;
361  triang->getVertexNeighbor(vSrc, i, vTgt);
362  q.push_back({vTgt, e});
363  }
364  };
365 
366  enqueueNeighbors(vBeg);
367  while(!q.empty()) {
368  const auto ve = q.back();
369  q.pop_back();
370  const auto v = ve.v;
371  const bool vIsAboveIso = inpScalars[v] > isoval;
372  if(vBegIsMin == vIsAboveIso) { // crossed the contour
373  xEdges.insert(ve.e);
374  } else {
375  innerVerts.push_back(v);
376  enqueueNeighbors(v);
377  }
378  }
379 
380  if(innerVerts.size() < _sizeMin)
381  return;
382  extendOutFld<scalarT>(xEdges, isoval, flag);
383  extendOutPts<scalarT>(innerVerts, isoval, flag, scalar);
384 }

387 template <typename scalarT, class Triang>
388 void ttk::ContourAroundPoint::extendOutFld(const std::set<SimplexId> &inpEdges,
389  float isoval,
390  int flag) const {
391  auto triang = reinterpret_cast<Triang *>(_inpFldTriang);
392  auto inpScalars = reinterpret_cast<const scalarT *>(_inpFldScalars);
393 
394  // Every edge e of the given edges maps to one output vertex v;
395  // v is connected to a number of other such vertices.
396  // Zero if e is part of an "antenna", one if e is part of the boundary
397  // of the grid, two in the default 2D case, possibly more in 3D.
398  // These neighbors of v are reached by looking at the triangles that e is a
399  // part of.
400 
401  std::map<SimplexId, SimplexId> inpe2outv;
402 
403  auto getOrMakeOutVert = [&](SimplexId e) {
404  if(inpe2outv.count(e))
405  return inpe2outv[e];
406 
407  // geometry of the vertex (linear interpolation on e)
408  const SimplexId v = _outContoursScalars.size();
409  inpe2outv[e] = v;
410 
411  SimplexId p;
412  triang->getEdgeVertex(e, 0, p);
413  SimplexId q;
414  triang->getEdgeVertex(e, 1, q);
415 
416  const float pVal = static_cast<float>(inpScalars[p]);
417  const float qVal = static_cast<float>(inpScalars[q]);
418  const double pFac = (qVal - isoval) / (qVal - pVal);
419  const double qFac = 1 - pFac;
420 
421  float px, py, pz;
422  triang->getVertexPoint(p, px, py, pz);
423  float qx, qy, qz;
424  triang->getVertexPoint(q, qx, qy, qz);
425  const float x = px * pFac + qx * qFac;
426  const float y = py * pFac + qy * qFac;
427  const float z = pz * pFac + qz * qFac;
428  _outContoursCoords.push_back(x);
429  _outContoursCoords.push_back(y);
430  _outContoursCoords.push_back(z);
431 
432  _outContoursScalars.push_back(isoval);
433  _outContoursFlags.push_back(flag);
434  return v;
435  };
436 
437  // Each input triangle can contribute to at most one output edge
438  std::set<SimplexId> handledTris;
439 
440  if(_inpDimMax == 3)
441  printWrn("Currently only the *edges* of contour surfaces are computed");
442 
443  for(const auto e : inpEdges) {
444  const auto nTriLoc = triang->getEdgeTriangleNumber(e);
445  if(nTriLoc == 0)
446  continue; // we do not output "contour points" (0D cells)
447 
448  const auto v = getOrMakeOutVert(e);
449 
450  for(SimplexId tLoc = 0; tLoc < nTriLoc; ++tLoc) {
451  SimplexId tGlo;
452  triang->getEdgeTriangle(e, tLoc, tGlo);
453  if(handledTris.count(tGlo))
454  continue;
455 
456  handledTris.insert(tGlo);
457  // Find the other edge that contains the isovalue
458  for(SimplexId eLoc = 0; eLoc < 3; ++eLoc) {
459  SimplexId eGlo;
460  triang->getTriangleEdge(tGlo, eLoc, eGlo);
461  if(eGlo == e || !inpEdges.count(eGlo))
462  continue;
463 
464  // new output edge from v(e) to v(eGlo)
465  _outContoursCinfos.push_back(2);
466  _outContoursCinfos.push_back(v);
467  _outContoursCinfos.push_back(getOrMakeOutVert(eGlo));
468  ++_outContoursNc;
469  } // END target edge
470  } // END triangle
471  } // END source edge
472 }

475 template <typename scalarT, class Triang>
476 void ttk::ContourAroundPoint::extendOutPts(
477  const std::vector<SimplexId> &vertices,
478  float isoval,
479  int flag,
480  float extremeVal) const {
481 
482  auto triang = reinterpret_cast<Triang *>(_inpFldTriang);
483  auto inpScalars = reinterpret_cast<const scalarT *>(_inpFldScalars);
484 
485  // We weight the vertices based on the difference to the extreme value.
486  // Vertices that are close to the extreme value get much weight,
487  // vertices that are close to the isovalue get little weight.
488  const double dMax = std::abs(isoval - extremeVal);
489 #ifndef NDEBUG
490  const double dMaxRelaxed = dMax * 1.001;
491 #endif
492 
493  // For now, we use a "full cosine" kernel.
494  const double kernelInputScaler = M_PI / dMax;
495  // d=0 ... cos(0)=1 ... w=1, d=dMax ... cos(pi)=-1 ... w = 0
496  auto k_func = [kernelInputScaler](double d) {
497  return (std::cos(d * kernelInputScaler) + 1.) / 2.;
498  };
499 
500  // do the computation in double precision
501  double wSum = 0.;
502  double outX = 0.;
503  double outY = 0.;
504  double outZ = 0.;
505  double outSca = 0.;
506 
507  for(const auto v : vertices) {
508  const scalarT vSca = inpScalars[v];
509  // Because we only use symmetric kernels for the weighting, the sign of
510  // the difference does not matter.
511  const double d = vSca - extremeVal;
512 #ifndef NDEBUG
513  if(std::abs(d) > dMaxRelaxed)
514  printWrn("d: " + std::to_string(d) + " dMax: " + std::to_string(dMax));
515 #endif
516  const double w = k_func(d);
517  wSum += w;
518 
519  float x, y, z;
520  triang->getVertexPoint(v, x, y, z);
521  outX += x * w;
522  outY += y * w;
523  outZ += z * w;
524  outSca += vSca * w;
525  }
526  assert(wSum != 0.);
527  outX /= wSum;
528  outY /= wSum;
529  outZ /= wSum;
530  outSca /= wSum;
531 
532  if(_radius > 0.) {
533  const double radiusCur = std::sqrt(outX * outX + outY * outY + outZ * outZ);
534  const double radiusScaler = _radius / radiusCur;
535  outX *= radiusScaler;
536  outY *= radiusScaler;
537  outZ *= radiusScaler;
538  }
539 
540  _outCentroidsCoords.push_back(static_cast<float>(outX));
541  _outCentroidsCoords.push_back(static_cast<float>(outY));
542  _outCentroidsCoords.push_back(static_cast<float>(outZ));
543  _outCentroidsScalars.push_back(static_cast<float>(outSca));
544  _outCentroidsFlags.push_back(flag);
545 }
