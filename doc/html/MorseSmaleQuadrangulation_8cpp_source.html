368 size_t ttk::MorseSmaleQuadrangulation::findSeparatrixMiddle(const size_t a,
369  const size_t b) {
370 
371  const int dim = 3;
372 
373  std::vector<float> distFromA(b - a + 1);
374  std::array<float, dim> prev{}, curr{};
375 
376  if(distFromA.empty()) {
377  return 0;
378  }
379 
380  curr[0] = sepPoints_[dim * a];
381  curr[1] = sepPoints_[dim * a + 1];
382  curr[2] = sepPoints_[dim * a + 2];
383 
384  // integrate distances at every point of this separatrix
385  for(size_t i = 1; i < b - a + 1; ++i) {
386  std::swap(curr, prev);
387  curr[0] = sepPoints_[dim * (a + i)];
388  curr[1] = sepPoints_[dim * (a + i) + 1];
389  curr[2] = sepPoints_[dim * (a + i) + 2];
390  distFromA[i]
391  = distFromA[i - 1] + ttk::Geometry::distance(&curr[0], &prev[0]);
392  }
393 
394  auto distAB = distFromA.back();
395  for(auto &el : distFromA) {
396  el = std::abs(el - distAB / 2.0);
397  }
398 
399  // index in separatrices point data array of separatrix middle
400  auto pos = a + std::min_element(distFromA.begin(), distFromA.end())
401  - distFromA.begin();
402 
403  // new point!
404  outputPoints_.emplace_back(sepPoints_[dim * pos]);
405  outputPoints_.emplace_back(sepPoints_[dim * pos + 1]);
406  outputPoints_.emplace_back(sepPoints_[dim * pos + 2]);
407 
408  SimplexId id = pos;
409 
410  // new point identifier (on the triangular mesh)
411  switch(sepCellDims_[pos]) {
412  case 0:
413  outputPointsIds_.emplace_back(sepCellIds_[pos]);
414  break;
415  case 1: {
416  // take the first vertex of the edge
417  triangulation_->getEdgeVertex(sepCellIds_[pos], 0, id);
418  outputPointsIds_.emplace_back(id);
419  break;
420  }
421  case 2: {
422  // take the first vertex of the triangle
423  triangulation_->getTriangleVertex(sepCellIds_[pos], 0, id);
424  outputPointsIds_.emplace_back(id);
425  break;
426  }
427  default:
428  break;
429  }
430 
431  outputPointsTypes_.emplace_back(1);
432 
433  return id;
434 }

647  std::vector<SimplexId> bounds{
648  criticalPointsIdentifier_[q.i], criticalPointsIdentifier_[q.j],
649  criticalPointsIdentifier_[q.k], criticalPointsIdentifier_[q.l]};

652  std::vector<bool> mask(morseSeg_.size(), false);
653  // restrict Dijkstra propagation to current cell
654  for(size_t j = 0; j < morseSeg_.size(); ++j) {
655  if(morseSeg_[j] == cellId_[i]) {
656  mask[j] = true;
657  }
658  }
659  // also allow to propagate on separatrices
660  for(const auto s : seps) {
661  for(size_t j = sepBegs_[s]; j <= sepEnds_[s]; ++j) {
662  if(sepCellDims_[j] == 1) {
663  auto e = sepCellIds_[j];

666  triangulation_->getEdgeVertex(e, 1, e1);
667  mask[e0] = true;
668  mask[e1] = true;
669  }
670  }
671  }
672 
673 #ifdef TTK_ENABLE_OPENMP
674 #pragma omp parallel for num_threads(threadNumber_)
675 #endif // TTK_ENABLE_OPENMP
676  for(size_t j = 0; j < outputDists.size(); ++j) {
677  Dijkstra::shortestPath(midsNearestVertex[j], *triangulation_,
678  outputDists.at(j), std::vector<SimplexId>(), mask);
679  }
680 
681  auto inf = std::numeric_limits<float>::infinity();
682  std::vector<float> sum(outputDists[0].size(), inf);
683 
684 #ifdef TTK_ENABLE_OPENMP
685 #pragma omp parallel for num_threads(threadNumber_)
686 #endif // TTK_ENABLE_OPENMP
687  for(size_t j = 0; j < sum.size(); ++j) {
688  // skip if vertex j not in cell i
689  if(morseSeg_[j] != cellId_[i]) {
690  continue;
691  }
692  auto m = outputDists[0][j];
693  auto n = outputDists[1][j];
694  auto o = outputDists[2][j];
695  auto p = outputDists[3][j];
696  if(m == inf || n == inf || o == inf || p == inf) {
697  continue;
698  }
699  // cost to minimize
700  sum[j] = m + n + o + p + std::abs(m - o) + std::abs(n - p);
701  }
702 
703  size_t verticesInCell
704  = verticesNumber_
705  - std::count(
706  sum.begin(), sum.end(), std::numeric_limits<float>::infinity());
707 
708  const size_t thresholdVertsInCell{50};
709  if(verticesInCell <= thresholdVertsInCell) {
710  std::stringstream msg;
711  msg << MODULE_S "Small cell detected" << std::endl;
712  dMsg(std::cout, msg.str(), infoMsg);
713  }
714 
715  SimplexId baryId{};
716  if(verticesInCell == 0) {
717  std::stringstream msg;
718  msg << MODULE_S "Barycenter in cell " << i << " not found." << std::endl;
719  dMsg(std::cout, msg.str(), infoMsg);
720 
721  // snap bary on sepMids[0]
722  baryId = outputPointsIds_[sepMids[0]];
723  } else {
724  baryId = std::min_element(sum.begin(), sum.end()) - sum.begin();
725  }
726 
727  long long baryPos = outputPointsIds_.size();

729  float x, y, z;
730  triangulation_->getVertexPoint(baryId, x, y, z);
731  outputPoints_.emplace_back(x);
732  outputPoints_.emplace_back(y);
733  outputPoints_.emplace_back(z);
734  outputPointsIds_.emplace_back(baryId);
735  outputPointsTypes_.emplace_back(2);
736  outputPointsCells_.emplace_back(i);
737  }
738 
739  std::vector<long long> srcs{};
740  std::vector<long long> dsts{};
741 
742  findSepsVertices(seps, srcs, dsts);
743 
744  auto sepsQuadVertex = [&](const size_t a, const size_t b) {
745  if(srcs[a] == srcs[b]) {
746  qsubd->emplace_back(Quad{4, srcs[a], sepMids[a], baryPos, sepMids[b]});
747  }
748  if(dsts[a] == dsts[b]) {
749  qsubd->emplace_back(Quad{4, dsts[a], sepMids[a], baryPos, sepMids[b]});
750  }
751  };
752 
753  // test every pair of current quad seps for a common vertex, if yes, use
754  // their middles and the quad barycenter
755  sepsQuadVertex(0, 1);
756  sepsQuadVertex(0, 2);
757  sepsQuadVertex(0, 3);
758  sepsQuadVertex(1, 2);
759  sepsQuadVertex(1, 3);
760  sepsQuadVertex(2, 3);
761  }
762 
763  subdiviseDegenerateQuads(outputSubd);
764 
765  // overwrite old quads
766  outputCells_ = std::move(outputSubd);
767 
768  return 0;
769 }

771 int ttk::MorseSmaleQuadrangulation::dualQuadrangulate() {
772 
773  // iterate over separatrices middles to build quadrangles around
774  // them: the separatrix vertices and two barycenters
775 
776  auto quads = reinterpret_cast<std::vector<Quad> *>(&outputCells_);
777  std::vector<long long> dualQuads{};
778  auto dquads = reinterpret_cast<std::vector<Quad> *>(&dualQuads);
779 
780  auto quadNeighbors = [&](const long long a) {
781  std::set<long long> neighs{};
782  for(const auto &q : *quads) {
783  if(a == q.i || a == q.k) {
784  neighs.emplace(q.j);
785  neighs.emplace(q.l);
786  }
787  if(a == q.j || a == q.l) {
788  neighs.emplace(q.i);
789  neighs.emplace(q.k);
790  }
791  }
792  return neighs;
793  };
794 
795  for(size_t i = 0; i < outputPointsIds_.size(); ++i) {
796  // only keep sep middles
797  if(outputPointsTypes_[i] != 1) {
798  continue;
799  }
800 
801  // get a list of direct neighbors
802  auto neighs = quadNeighbors(i);
803 
804  // skip sep middle between single extremum and saddle in
805  // degenerate cell (point not used)
806  if(neighs.empty()) {
807  continue;
808  }
809 
810  // neighs should contain exactly 4 indices, two corresponding to
811  // critical points, and two for generated points, mostly cell
812  // barycenters
813  std::vector<long long> crit{}, gen{};
814  for(const auto n : neighs) {
815  if(n < criticalPointsNumber_) {
816  crit.emplace_back(n);
817  } else {
818  gen.emplace_back(n);
819  }
820  }
821 
822  dquads->emplace_back(Quad{4, crit[0], gen[0], crit[1], gen[1]});
823  }
824 
825  // for degenerate quadrangles, iterate over the single extremum, and
826  // use the three generated points v0, v1 and v2
827  for(size_t i = 0; i < outputPointsIds_.size(); ++i) {
828 
829  // only keep v0 points in degenerate cells
830  if(outputPointsTypes_[i] != 3) {
831  continue;
832  }
833 
834  // v0 has 3 neighbors: v1, v2 and the double extremum
835 
836  auto v0neighs = quadNeighbors(i);
837  std::vector<long long> crit{}, gen{};
838  for(const auto n : v0neighs) {
839  if(n < criticalPointsNumber_) {
840  crit.emplace_back(n);
841  } else {
842  gen.emplace_back(n);
843  }
844  }
845 
846  // v1 has 3 neighbors: v0, a sep middle and the single extremum
847 
848  // follow v1 to get the single extremum
849  auto v1neighs = quadNeighbors(gen[0]);
850  long long single{};
851  for(const auto n : v1neighs) {
852  if(n < criticalPointsNumber_) {
853  single = n;
854  }
855  }
856 
857  // the single extremum has 3 neighbors: v1, v2 and the saddle
858 
859  // follow single to find saddle
860  auto sneighs = quadNeighbors(single);
861  for(const auto n : sneighs) {
862  if(n < criticalPointsNumber_) {
863  crit.emplace_back(n);
864  }
865  }
866 
867  dquads->emplace_back(Quad{4, crit[0], gen[0], crit[1], gen[1]});
868  }
869 
870  outputCells_ = std::move(dualQuads);
871 
872  return 0;
873 }

875 bool ttk::MorseSmaleQuadrangulation::checkSurfaceCloseness() const {
876  bool triangulationClosed{true};
877  // sweep over all vertices to check if one is on a boundary
878  for(SimplexId i = 0; i < verticesNumber_; ++i) {
879  if(triangulation_->isVertexOnBoundary(i)) {
880  triangulationClosed = false;
881  break;
882  }
883  }
884  // quadrangles edges -> quadrangles
885  std::map<std::pair<long long, long long>, std::set<size_t>> quadEdges{};
886 
887  // sweep over quadrangulation edges
888  auto quads = reinterpret_cast<const std::vector<Quad> *>(&outputCells_);
889  for(size_t i = 0; i < quads->size(); ++i) {
890  auto q = quads->at(i);
891  // store edges in order
892  if(q.i < q.j) {
893  quadEdges[std::make_pair(q.i, q.j)].emplace(i);
894  } else {
895  quadEdges[std::make_pair(q.j, q.i)].emplace(i);
896  }
897  if(q.j < q.k) {
898  quadEdges[std::make_pair(q.j, q.k)].emplace(i);
899  } else {
900  quadEdges[std::make_pair(q.k, q.j)].emplace(i);
901  }
902  if(q.k < q.l) {
903  quadEdges[std::make_pair(q.k, q.l)].emplace(i);
904  } else {
905  quadEdges[std::make_pair(q.l, q.k)].emplace(i);
906  }
907  if(q.l < q.i) {
908  quadEdges[std::make_pair(q.l, q.i)].emplace(i);
909  } else {
910  quadEdges[std::make_pair(q.i, q.l)].emplace(i);
911  }
912  }
913 
914  bool quadrangulationClosed{true};
915  for(const auto &e : quadEdges) {
916  if(e.second.size() < 2) {
917  quadrangulationClosed = false;
918  break;
919  }
920  }
921 
922  // each edge should be shared by two different quadrangles
923  return triangulationClosed == quadrangulationClosed;
924 }
