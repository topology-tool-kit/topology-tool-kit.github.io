245  y[i] = 0;
246  crit_coords_x[i] = 0;
247  crit_coords_y[i] = 0;
248  crit_coords_z[i] = 0;

252  min_prices[j] = std::numeric_limits<dataType>::max();

254 
255  std::vector<Bidder<dataType> *>
256  points_to_append; // Will collect bidders linked to diagonal
257  // 2. Preprocess the matchings
258  for(unsigned int j = 0; j < matchings.size(); j++) {
259  for(unsigned int i = 0; i < matchings[j].size(); i++) {
260  int bidder_id = std::get<0>(matchings[j][i]);
261  int good_id = std::get<1>(matchings[j][i]);
262  if(good_id < 0 && bidder_id >= 0) {
263  // Future new barycenter point
264  points_to_append.push_back(&current_bidder_diagrams_[j].get(bidder_id));
265  }
266 
267  else if(good_id >= 0 && bidder_id >= 0) {
268  // Update coordinates (to be divided by the number of diagrams later on)
269  x[good_id] += current_bidder_diagrams_[j].get(bidder_id).x_;
270  y[good_id] += current_bidder_diagrams_[j].get(bidder_id).y_;
271  if(geometrical_factor_ < 1) {
272  std::tuple<float, float, float> critical_coordinates
273  = current_bidder_diagrams_[j]
274  .get(bidder_id)
275  .GetCriticalCoordinates();
276  crit_coords_x[good_id] += std::get<0>(critical_coordinates);
277  crit_coords_y[good_id] += std::get<1>(critical_coordinates);
278  crit_coords_z[good_id] += std::get<2>(critical_coordinates);
279  }
280  } else if(good_id >= 0 && bidder_id < 0) {
281  // Counting the number of times this barycenter point is linked to the
282  // diagonal
283  count_diag_matchings[good_id] = count_diag_matchings[good_id] + 1;
284  }
285  }
286  }
287 
288  // 3. Update the previous points of the barycenter
289  for(unsigned int i = 0; i < n_goods; i++) {
290  if(count_diag_matchings[i] < n_diagrams) {
291  // Barycenter point i is matched at least to one off-diagonal bidder
292  // 3.1 Compute the arithmetic mean of off-diagonal bidders linked to it
293  dataType x_bar = x[i] / (double)(n_diagrams - count_diag_matchings[i]);
294  dataType y_bar = y[i] / (double)(n_diagrams - count_diag_matchings[i]);
295  // 3.2 Compute the new coordinates of the point (the more linked to the
296  // diagonal it was, the closer to the diagonal it'll be)
297  dataType new_x
298  = ((double)(n_diagrams - count_diag_matchings[i]) * x_bar
299  + (double)count_diag_matchings[i] * (x_bar + y_bar) / 2.)
300  / (double)n_diagrams;
301  dataType new_y
302  = ((double)(n_diagrams - count_diag_matchings[i]) * y_bar
303  + (double)count_diag_matchings[i] * (x_bar + y_bar) / 2.)
304  / (double)n_diagrams;
305  // TODO Weight by persistence
306  dataType new_crit_coord_x
307  = crit_coords_x[i] / (double)(n_diagrams - count_diag_matchings[i]);
308  dataType new_crit_coord_y
309  = crit_coords_y[i] / (double)(n_diagrams - count_diag_matchings[i]);
310  dataType new_crit_coord_z
311  = crit_coords_z[i] / (double)(n_diagrams - count_diag_matchings[i]);
312 
313  // 3.3 Compute and store how much the point has shifted
314  // TODO adjust shift with geometrical_factor_
315  dataType dx = barycenter_goods_[0].get(i).x_ - new_x;
316  dataType dy = barycenter_goods_[0].get(i).y_ - new_y;
317  dataType shift = Geometry::pow(abs(dx), wasserstein_)
318  + Geometry::pow(abs(dy), wasserstein_);
319  if(shift > max_shift) {
320  max_shift = shift;
321  }
322  // 3.4 Update the position of the point
323  for(unsigned int j = 0; j < n_diagrams; j++) {
324  barycenter_goods_[j].get(i).SetCoordinates(new_x, new_y);
325  if(geometrical_factor_ < 1) {
326  barycenter_goods_[j].get(i).SetCriticalCoordinates(
327  new_crit_coord_x, new_crit_coord_y, new_crit_coord_z);
328  }
329  if(barycenter_goods_[j].get(i).getPrice() < min_prices[j]) {
330  min_prices[j] = barycenter_goods_[j].get(i).getPrice();
331  }
332  }
333  // TODO Reinitialize/play with prices here if you wish
334  }
335  }
336  for(unsigned int j = 0; j < n_diagrams; j++) {
337  if(min_prices[j] >= std::numeric_limits<dataType>::max() / 2.) {
338  min_prices[j] = 0;
339  }
340  }
341 
342  // 4. Delete off-diagonal barycenter points not linked to any
343  // off-diagonal bidder
344  for(unsigned int i = 0; i < n_goods; i++) {
345  if(count_diag_matchings[i] == n_diagrams) {
346  points_deleted_ += 1;
347  dataType shift
348  = 2
349  * Geometry::pow(
350  barycenter_goods_[0].get(i).getPersistence() / 2., wasserstein_);
351  if(shift > max_shift) {
352  max_shift = shift;
353  }
354  for(unsigned int j = 0; j < n_diagrams; j++) {
355  barycenter_goods_[j].get(i).id_ = -1;
356  }
357  }
358  }
359 
360  // 5. Append the new points to the barycenter
361  for(unsigned int k = 0; k < points_to_append.size(); k++) {
362  points_added_ += 1;
363  Bidder<dataType> *b = points_to_append[k];
364  dataType gx
365  = (b->x_ + (n_diagrams - 1) * (b->x_ + b->y_) / 2.) / (n_diagrams);
366  dataType gy
367  = (b->y_ + (n_diagrams - 1) * (b->x_ + b->y_) / 2.) / (n_diagrams);
368  std::tuple<dataType, dataType, dataType> critical_coordinates
369  = b->GetCriticalCoordinates();
370  for(unsigned int j = 0; j < n_diagrams; j++) {
371  Good<dataType> g
372  = Good<dataType>(gx, gy, false, barycenter_goods_[j].size());
373  g.setPrice(min_prices[j]);
374  if(geometrical_factor_ < 1) {
375  g.SetCriticalCoordinates(std::get<0>(critical_coordinates),
376  std::get<1>(critical_coordinates),
377  std::get<2>(critical_coordinates));
378  }
379  barycenter_goods_[j].addGood(g);
380  dataType shift
381  = 2
382  * Geometry::pow(barycenter_goods_[j].get(g.id_).getPersistence() / 2.,
383  wasserstein_);
384  if(shift > max_shift) {
385  max_shift = shift;
386  }
387  }
388  }
389 
390  // 6. Finally, recreate barycenter_goods
391  for(unsigned int j = 0; j < n_diagrams; j++) {
392  int count = 0;
393  GoodDiagram<dataType> new_barycenter;
394  for(int i = 0; i < barycenter_goods_[j].size(); i++) {
395  Good<dataType> g = barycenter_goods_[j].get(i).copy();
396  if(g.id_ != -1) {
397  g.id_ = count;
398  new_barycenter.addGood(g);
399  count++;
400  }
401  }
402  barycenter_goods_[j] = new_barycenter;
403  }
404 
405  // std::cout<<"TIME OF UPDATE : "<< t_update.getElapsedTime()<<std::endl;
406  return max_shift;

644 template <typename dataType>
645 typename PDBarycenter<dataType>::KDTreePair
646  PDBarycenter<dataType>::getKDTree() const {
647  Timer tm;
648  auto kdt = std::unique_ptr<KDTree<dataType>>(
649  new KDTree<dataType>{true, wasserstein_});
650 
651  const int dimension = geometrical_factor_ >= 1 ? 2 : 5;
652 
653  std::vector<dataType> coordinates;
654  std::vector<std::vector<dataType>> weights;
655 
656  for(int i = 0; i < barycenter_goods_[0].size(); i++) {
657  const Good<dataType> &g = barycenter_goods_[0].get(i);
658  coordinates.push_back(geometrical_factor_ * g.x_);
659  coordinates.push_back(geometrical_factor_ * g.y_);
660  if(geometrical_factor_ < 1) {
661  coordinates.push_back((1 - geometrical_factor_) * g.coords_x_);
662  coordinates.push_back((1 - geometrical_factor_) * g.coords_y_);
663  coordinates.push_back((1 - geometrical_factor_) * g.coords_z_);
664  }
665  }
666 
667  for(unsigned int idx = 0; idx < barycenter_goods_.size(); idx++) {
668  std::vector<dataType> empty_weights;
669  weights.push_back(empty_weights);
670  for(int i = 0; i < barycenter_goods_[idx].size(); i++) {
671  const Good<dataType> &g = barycenter_goods_[idx].get(i);
672  weights[idx].push_back(g.getPrice());
673  }
674  }
675  // Correspondance map : position in barycenter_goods_ --> KDT node
676 
677  auto correspondance_kdt_map
678  = kdt->build(coordinates.data(), barycenter_goods_[0].size(), dimension,
679  weights, barycenter_goods_.size());
680  this->printMsg(" Building KDTree", 1, tm.getElapsedTime(),
681  debug::LineMode::NEW, debug::Priority::VERBOSE);
682  return std::make_pair(std::move(kdt), correspondance_kdt_map);
683 }
