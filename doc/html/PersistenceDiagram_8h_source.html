262 int ttk::PersistenceDiagram::executePersistentSimplex(
263  std::vector<PersistencePair> &CTDiagram,
264  const scalarType *inputScalars,
265  const SimplexId *inputOffsets,
266  const triangulationType *triangulation) {
267 
268  Timer tm{};
269  const auto dim = triangulation->getDimensionality();
270 
271  std::vector<ttk::PersistentSimplexPairs::PersistencePair> pairs{};
272 
273  dcg_.setInputOffsets(inputOffsets);
274  psp_.setDebugLevel(this->debugLevel_);
275  psp_.setThreadNumber(this->threadNumber_);
276  psp_.computePersistencePairs(pairs, inputOffsets, *triangulation);
277 
278  // convert PersistentSimplex pairs (with critical cells id) to PL
279  // pairs (with vertices id)
280 
281  for(auto &p : pairs) {
282  int birthType{};
283  if(dim == 3) {
284  birthType = p.type;
285  } else if(dim == 2) {
286  birthType = (p.type == 0) ? 0 : 1;
287  }
288  if(p.type > 0) {
289  p.birth
290  = dcg_.getCellGreaterVertex(Cell{birthType, p.birth}, *triangulation);
291  }
292  if(p.death != -1) {
293  p.death = dcg_.getCellGreaterVertex(
294  Cell{birthType + 1, p.death}, *triangulation);
295  }
296  }
297 
298  CTDiagram.reserve(pairs.size() + 1);
299 
300  // find the global maximum
301  const auto nVerts = triangulation->getNumberOfVertices();
302  const auto globmax = std::distance(
303  inputOffsets, std::max_element(inputOffsets, inputOffsets + nVerts));
304 
305  // convert pairs to the relevant format
306  for(const auto &p : pairs) {
307  const auto isFinite = (p.death >= 0);
308  const auto death = isFinite ? p.death : globmax;
309  if(p.type == 0) {
310  CTDiagram.emplace_back(
311  p.birth, CriticalType::Local_minimum, death,
312  (isFinite && p.type < dim - 1) ? CriticalType::Saddle1
313  : CriticalType::Local_maximum,
314  inputScalars[death] - inputScalars[p.birth], p.type);
315  } else if(p.type == 1) {
316  CTDiagram.emplace_back(
317  p.birth, CriticalType::Saddle1, death,
318  (isFinite && dim == 3) ? CriticalType::Saddle2
319  : CriticalType::Local_maximum,
320  inputScalars[death] - inputScalars[p.birth], p.type);
321  } else if(p.type == 2) {
322  CTDiagram.emplace_back(
323  p.birth, CriticalType::Saddle2, death, CriticalType::Local_maximum,
324  inputScalars[death] - inputScalars[p.birth], p.type);
325  }
326  }
327 
328  this->printMsg("Complete", 1.0, tm.getElapsedTime(), this->threadNumber_);
329  return 0;
330 }

373 template <typename scalarType, class triangulationType>
374 int ttk::PersistenceDiagram::executeFTM(
375  std::vector<PersistencePair> &CTDiagram,
376  const scalarType *inputScalars,
377  const SimplexId *inputOffsets,
378  const triangulationType *triangulation) {
379 
380  contourTree_.setVertexScalars(inputScalars);
381  contourTree_.setTreeType(ftm::TreeType::Join_Split);
382  contourTree_.setVertexSoSoffsets(inputOffsets);
383  contourTree_.setSegmentation(false);
384  contourTree_.build<scalarType>(triangulation);
385 
386  // get persistence pairs
387  std::vector<std::tuple<ttk::SimplexId, ttk::SimplexId, scalarType>> JTPairs;
388  std::vector<std::tuple<ttk::SimplexId, ttk::SimplexId, scalarType>> STPairs;
389  contourTree_.computePersistencePairs<scalarType>(JTPairs, true);
390  contourTree_.computePersistencePairs<scalarType>(STPairs, false);
391 
392  // merge pairs
393  const auto JTSize = JTPairs.size();
394  const auto STSize = STPairs.size();
395  std::vector<std::tuple<ttk::SimplexId, ttk::SimplexId, scalarType, bool>>
396  CTPairs(JTSize + STSize);
397  for(size_t i = 0; i < JTSize; ++i) {
398  const auto &x = JTPairs[i];
399  CTPairs[i]
400  = std::make_tuple(std::get<0>(x), std::get<1>(x), std::get<2>(x), true);
401  }
402  for(size_t i = 0; i < STSize; ++i) {
403  const auto &x = STPairs[i];
404  CTPairs[JTSize + i]
405  = std::make_tuple(std::get<0>(x), std::get<1>(x), std::get<2>(x), false);
406  }
407 
408  // remove the last pair which is present two times (global extrema pair)
409  if(!CTPairs.empty()) {
410  auto cmp =
411  [](
412  const std::tuple<ttk::SimplexId, ttk::SimplexId, scalarType, bool> &a,
413  const std::tuple<ttk::SimplexId, ttk::SimplexId, scalarType, bool> &b) {
414  return std::get<2>(a) < std::get<2>(b);
415  };
416 
417  std::sort(CTPairs.begin(), CTPairs.end(), cmp);
418  CTPairs.erase(CTPairs.end() - 1);
419  }
420 
421  // get persistence diagrams
422  computeCTPersistenceDiagram<scalarType>(contourTree_, CTPairs, CTDiagram);
423 
424  // get the saddle-saddle pairs
425  std::vector<std::tuple<SimplexId, SimplexId, scalarType>>
426  pl_saddleSaddlePairs;
427  if(triangulation->getDimensionality() == 3 and ComputeSaddleConnectors) {
428  dcg_.setInputScalarField(inputScalars);
429  dcg_.setInputOffsets(inputOffsets);
430  dcg_.computeSaddleSaddlePersistencePairs<scalarType>(
431  pl_saddleSaddlePairs, *triangulation);
432 
433  // add saddle-saddle pairs to the diagram
434  for(const auto &i : pl_saddleSaddlePairs) {
435  const ttk::SimplexId v0 = std::get<0>(i);
436  const ttk::SimplexId v1 = std::get<1>(i);
437  const auto persistenceValue = static_cast<double>(std::get<2>(i));
438 
439  CTDiagram.emplace_back(v0, ttk::CriticalType::Saddle1, v1,
440  ttk::CriticalType::Saddle2, persistenceValue, 1);
441  }
442  }
443  return 0;
444 }
