399  // Check the consistency of the variables
400 #ifndef TTK_ENABLE_KAMIKAZE
401  if(!numberOfInputs_)
402  return -1;
403  if(!vertexNumber_)
404  return -2;
405 
406  for(int i = 0; i < numberOfInputs_; i++) {
407  if(!inputData_[i])
408  return -4;
409  }
410  if(!outputLowerBoundField_)
411  return -5;
412  if(!outputUpperBoundField_)
413  return -6;
414 #endif
415 
416  SimplexId count = 0;
417 
418  // Pointers type casting
419  dataType *outputLowerBoundField = (dataType *)outputLowerBoundField_;
420  dataType *outputUpperBoundField = (dataType *)outputUpperBoundField_;
421  dataType **inputData = (dataType **)inputData_.data();
422  double *outputMeanField = static_cast<double *>(outputMeanField_);
423 
424 #ifdef TTK_ENABLE_OPENMP
425 #pragma omp parallel for num_threads(threadNumber_)
426 #endif
427 
428  for(SimplexId v = 0; v < (SimplexId)vertexNumber_; v++) {
429 
430  // Avoid any processing if the abort signal is sent
431  if((!wrapper_) || ((wrapper_) && (!wrapper_->needsToAbort()))) {
432 
433  // For the lower bound scalar field
434  if(computeLowerBound_) {
435  // Initialisation : values of the first input
436  outputLowerBoundField[v] = inputData[0][v];
437  // Loop over the inputs
438  for(int inp = 1; inp < numberOfInputs_; inp++) {
439  // Minimum value
440  if(computeLowerBound_)
441  if(inputData[inp][v] < outputLowerBoundField[v])
442  outputLowerBoundField[v] = inputData[inp][v];
443  }
444  }
445 
446  // For the upper bound scalar field
447  if(computeUpperBound_) {
448  // Initialisation : values of the first input
449  outputUpperBoundField[v] = inputData[0][v];
450  // Loop over the inputs
451  for(int inp = 1; inp < numberOfInputs_; inp++) {
452  // Maximum value
453  if(computeUpperBound_)
454  if(inputData[inp][v] > outputUpperBoundField[v])
455  outputUpperBoundField[v] = inputData[inp][v];
456  }
457  }
458 
459  // Update the progress bar of the wrapping code -- to adapt
460  if(debugLevel_ > advancedInfoMsg) {
461 #ifdef TTK_ENABLE_OPENMP
462 #pragma omp critical
463 #endif
464  {
465  if((wrapper_) && (!(count % ((vertexNumber_) / 10)))) {
466  wrapper_->updateProgress((count + 1.0) / vertexNumber_);
467  }
468 
469  count++;
470  }
471  }
472  }
473  }
474 
475  // Histogram
476  if(computeUpperBound_ && computeLowerBound_) {
477  // Range
478  double range[2];
479  range[0] = outputLowerBoundField[0];
480  range[1] = outputUpperBoundField[0];
481 
482  for(SimplexId v = 0; v < vertexNumber_; v++) {
483  if(outputLowerBoundField[v] < range[0])
484  range[0] = outputLowerBoundField[v];
485  if(outputUpperBoundField[v] > range[1])
486  range[1] = outputUpperBoundField[v];
487  }
488 
489  // Interval between bins
490  double dx = (range[1] - range[0]) / (double)binCount_;
491 
492  // Bin values
493  for(int b = 0; b < binCount_; b++) {
494  binValues_[b] = range[0] + (dx / 2.0) + (double)b * dx;
495  }
496 
497  int idx;
498  double increment = 1.0 / (double)numberOfInputs_;
499 #ifdef TTK_ENABLE_OPENMP
500 #pragma omp parallel for private(idx) num_threads(threadNumber_)
501 #endif
502  for(SimplexId v = 0; v < vertexNumber_; v++) {
503  for(int i = 0; i < numberOfInputs_; i++) {
504  idx = (int)floor((inputData[i][v] - range[0]) * binCount_
505  / (range[1] - range[0]));
506  idx = (idx == binCount_) ? binCount_ - 1 : idx;
507  outputProbability_[idx][v] += increment;
508  }
509  }
510  }
511 
512  // Mean field
513  for(SimplexId v = 0; v < vertexNumber_; v++) {
514  double sum = 0.0;
515  for(int i = 0; i < numberOfInputs_; i++) {
516  sum += static_cast<double>(inputData[i][v]);
517  }
518  outputMeanField[v] = sum / static_cast<double>(numberOfInputs_);
519  }
520 
521  {
522  std::stringstream msg;
523  msg << "[UncertainDataEstimator] Data-set (" << vertexNumber_
524  << " points) processed in " << t.getElapsedTime() << " s. ("
525  << threadNumber_ << " thread(s))." << std::endl;
526  dMsg(std::cout, msg.str(), timeMsg);
527  }
528 
529  return 0;
