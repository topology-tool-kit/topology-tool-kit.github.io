314  int min_dim = std::min(nRows, nCols);
315 
316  // --- Construct all possible assignments

318  // We hard write the basic assignments to avoid the call of

321  if(min_dim == 1 and max_dim == 1)

339  allAsgn = constructAssignments3(min_dim, max_dim);
