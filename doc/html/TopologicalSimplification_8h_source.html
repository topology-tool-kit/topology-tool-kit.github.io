137 template <typename triangulationType>
138 int ttk::TopologicalSimplification::getCriticalType(
139  SimplexId vertex,
140  const SimplexId *const offsets,
141  const triangulationType &triangulation) const {
142 
143  bool isMinima{true};
144  bool isMaxima{true};
145  SimplexId neighborNumber = triangulation.getVertexNeighborNumber(vertex);
146  for(SimplexId i = 0; i < neighborNumber; ++i) {
147  SimplexId neighbor;
148  triangulation.getVertexNeighbor(vertex, i, neighbor);
149 
150  if(offsets[neighbor] < offsets[vertex])
151  isMinima = false;
152  if(offsets[neighbor] > offsets[vertex])
153  isMaxima = false;
154  if(!isMinima and !isMaxima) {
155  return 0;
156  }
157  }
158 
159  if(isMinima)
160  return -1;
161  if(isMaxima)
162  return 1;
163 
164  return 0;
165 }

217 template <typename dataType>
218 int ttk::TopologicalSimplification::addPerturbation(
219  dataType *const scalars, SimplexId *const offsets) const {
220  dataType epsilon{};
221 
222  if(std::is_same<dataType, double>::value)
223  epsilon = Geometry::powIntTen<dataType>(1 - DBL_DIG);
224  else if(std::is_same<dataType, float>::value)
225  epsilon = Geometry::powIntTen<dataType>(1 - FLT_DIG);
226  else
227  return -1;
228 
229  std::vector<std::tuple<dataType, SimplexId, SimplexId>> perturbation(
230  vertexNumber_);
231  for(SimplexId i = 0; i < vertexNumber_; ++i) {
232  std::get<0>(perturbation[i]) = scalars[i];
233  std::get<1>(perturbation[i]) = offsets[i];
234  std::get<2>(perturbation[i]) = i;
235  }
236 
237  SweepCmp cmp(true);
238  sort(perturbation.begin(), perturbation.end(), cmp);
239 
240  for(SimplexId i = 0; i < vertexNumber_; ++i) {
241  if(i) {
242  if(std::get<0>(perturbation[i]) <= std::get<0>(perturbation[i - 1]))
243  std::get<0>(perturbation[i])
244  = std::get<0>(perturbation[i - 1]) + epsilon;
245  }
246  scalars[std::get<2>(perturbation[i])] = std::get<0>(perturbation[i]);
247  }
248 
249  return 0;
250 }
