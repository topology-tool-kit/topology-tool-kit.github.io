136 template <typename triangulationType>
137 inline Point
138 relax(const SimplexId a,
139 std::vector<ttk::SurfaceGeometrySmoother::Point> &outputPoints,
140 const triangulationType &triangulationToSmooth) const {
141 Point relaxed{outputPoints[a]};
142 const auto nneigh{triangulationToSmooth.getVertexNeighborNumber(a)};
143 for(SimplexId i = 0; i < nneigh; ++i) {
144 SimplexId neigh{};
145 triangulationToSmooth.getVertexNeighbor(a, i, neigh);
146 relaxed = relaxed + outputPoints[neigh];
147 }
148 return relaxed * (1.0F / static_cast<float>(nneigh + 1));
149 }

159 inline Point projectOnTrianglePlane(const Point &p,
160 const Point &a,
161 const Point &normTri) const {
162 const auto ap{p - a};
163 return p - normTri * Geometry::dotProduct(normTri.data(), ap.data());
164 }
165
174 inline Point
175 projectOnEdge(const Point &p, const Point &a, const Point &b) const {
176 const auto ab{b - a};
177 const auto ap{p - a};
178 return a
179 + ab * Geometry::dotProduct(ap.data(), ab.data())
180 / Geometry::dotProduct(ab.data(), ab.data());
181 }

191 template <typename triangulationType>
192 inline SimplexId
193 getNearestSurfaceVertex(const Point &pa,
194 std::vector<float> &dists,
195 const triangulationType &triangulation) const {
196 for(SimplexId i = 0; i < triangulation.getNumberOfVertices(); ++i) {
197 Point pv{};
198 triangulation.getVertexPoint(i, pv[0], pv[1], pv[2]);
199 dists[i] = Geometry::distance(pa.data(), pv.data());
200 }
201 return std::min_element(dists.begin(), dists.end()) - dists.begin();
202 }

207template <typename triangulationType>
208ttk::SurfaceGeometrySmoother::ProjectionResult
209 ttk::SurfaceGeometrySmoother::findProjection(
210 const ProjectionInput &pi,
211 VisitedMask &trianglesTested,
212 std::vector<float> &dists,
213 std::stack<SimplexId> &trianglesToTest,
214 const triangulationType &triangulation) const {
215
216 ProjectionResult res{pi.pt, pi.nearestVertex, 0, false};
217
218 // clean trianglesToTest
219 while(!trianglesToTest.empty()) {
220 trianglesToTest.pop();
221 }
222
223 // init pipeline by checking in the first triangle around selected vertex
224 if(triangulation.getVertexTriangleNumber(res.nearestVertex) > 0) {
225 SimplexId next{};
226 triangulation.getVertexTriangle(res.nearestVertex, 0, next);
227 trianglesToTest.push(next);
228 }
229
230 while(!trianglesToTest.empty()) {
231 const auto curr = trianglesToTest.top();
232 trianglesToTest.pop();
233
234 // skip if already tested
235 if(trianglesTested.isVisited_[curr]) {
236 continue;
237 }
238
239 // get triangle vertices
240 std::array<SimplexId, 3> tverts{};
241 triangulation.getTriangleVertex(curr, 0, tverts[0]);
242 triangulation.getTriangleVertex(curr, 1, tverts[1]);
243 triangulation.getTriangleVertex(curr, 2, tverts[2]);
244
245 // get coordinates of triangle vertices (lets name is MNO)
246 std::array<Point, 3> mno{}; // [0] is M, [1] is N and [2] is O
247 triangulation.getVertexPoint(tverts[0], mno[0][0], mno[0][1], mno[0][2]);
248 triangulation.getVertexPoint(tverts[1], mno[1][0], mno[1][1], mno[1][2]);
249 triangulation.getVertexPoint(tverts[2], mno[2][0], mno[2][1], mno[2][2]);
250
251 // triangle normal: cross product of two edges
252 Point crossP{};
253 // mn, mo vectors
254 const Point mn = mno[1] - mno[0];
255 const Point mo = mno[2] - mno[0];
256 // compute mn ^ mo
257 Geometry::crossProduct(mn.data(), mo.data(), crossP.data());
258 // unitary normal vector
259 const Point normTri = crossP / Geometry::magnitude(crossP.data());
260
261 res.pt = this->projectOnTrianglePlane(pi.pt, mno[0], normTri);
262
263 // compute barycentric coords of projection
264 Point baryCoords{};
265 Geometry::computeBarycentricCoordinates(
266 mno[0].data(), mno[1].data(), mno[2].data(), res.pt.data(), baryCoords);
267
268 // check if projection in triangle
269 bool inTriangle = true;
270 static const float PREC_FLT{powf(10, -FLT_DIG)};
271
272 for(auto &coord : baryCoords) {
273 if(coord < -PREC_FLT) {
274 inTriangle = false;
275 }
276 if(coord > 1 + PREC_FLT) {
277 inTriangle = false;
278 }
279 }
280
281 // mark triangle as tested
282 trianglesTested.insert(curr);
283 res.trianglesChecked++;
284
285 if(inTriangle) {
286 res.projSuccess = true;
287 // should we check if we have the nearest triangle?
288 break;
289 }
290
291 // extrema values in baryCoords
292 const auto extrema
293 = std::minmax_element(baryCoords.begin(), baryCoords.end());
294
295 // find the nearest triangle vertices local ids (with the
296 // highest/positive values in baryCoords) from proj
297 std::array<SimplexId, 2> vid{
298 static_cast<SimplexId>(extrema.second - baryCoords.begin()), 0};
299 for(size_t j = 0; j < baryCoords.size(); j++) {
300 if(j != static_cast<size_t>(extrema.first - baryCoords.begin())
301 && j != static_cast<size_t>(extrema.second - baryCoords.begin())) {
302 vid[1] = j;
303 break;
304 }
305 }
306
307 // store vertex with highest barycentric coordinate
308 res.nearestVertex = tverts[vid[0]];
309 const auto secondNearVert{tverts[vid[1]]};
310
311 // get the triangle edge with the two vertices
312 SimplexId edge{};
313 const auto nEdges{triangulation.getVertexEdgeNumber(res.nearestVertex)};
314 for(SimplexId i = 0; i < nEdges; ++i) {
315 triangulation.getVertexEdge(res.nearestVertex, i, edge);
316 SimplexId v{};
317 triangulation.getEdgeVertex(edge, 0, v);
318 if(v == res.nearestVertex) {
319 triangulation.getEdgeVertex(edge, 1, v);
320 }
321 if(v == secondNearVert) {
322 break;
323 }
324 }
325
326 // next triangle to visit
327 SimplexId next{};
328 const auto nTri{triangulation.getEdgeTriangleNumber(edge)};
329 for(SimplexId i = 0; i < nTri; ++i) {
330 triangulation.getEdgeTriangle(edge, i, next);
331 if(next != curr) {
332 break;
333 }
334 }
335
336 const auto nVisited{trianglesTested.visitedIds_.size()};
337 if(nVisited > 1 && next == trianglesTested.visitedIds_[nVisited - 2]) {
338 // the relaxed point is probably over the edge separating the
339 // current and the last visited triangles
340 res.pt = this->projectOnEdge(pi.pt, mno[vid[0]], mno[vid[1]]);
341 res.projSuccess = true;
342 break;
343 }
344
345 if(!trianglesTested.isVisited_[next]) {
346 trianglesToTest.push(next);
347 }
348 }
349
350 const size_t maxTrChecked = 100;
351
352 if(res.projSuccess && res.trianglesChecked > maxTrChecked) {
353 res.projSuccess = false;
354 }
355
356 if(!res.projSuccess) {
357 // replace proj by the nearest vertex?
358 res.nearestVertex
359 = this->getNearestSurfaceVertex(pi.pt, dists, triangulation);
360 triangulation.getVertexPoint(
361 res.nearestVertex, res.pt[0], res.pt[1], res.pt[2]);
362 }
363
364 return res;
365}

367template <typename triangulationType0, typename triangulationType1>
368int ttk::SurfaceGeometrySmoother::relaxProject(
369 std::vector<ttk::SurfaceGeometrySmoother::Point> &outputPoints,
370 std::vector<ttk::SurfaceGeometrySmoother::Point> &tmpStorage,
371 std::vector<SimplexId> &nearestVertexId,
372 std::vector<bool> &trianglesTested,
373 std::vector<SimplexId> &visitedTriangles,
374 std::vector<float> &dists,
375 const char *const mask,
376 const triangulationType0 &triangulationToSmooth,
377 const triangulationType1 &triangulationSurface) const {
378
379 Timer tm;
380 std::stack<SimplexId> trianglesToTest{};
381
382 // main loop
383#ifdef TTK_ENABLE_OPENMP
384#pragma omp parallel for num_threads(threadNumber_) \
385 firstprivate(trianglesTested, visitedTriangles, dists, trianglesToTest)
386#endif // TTK_ENABLE_OPENMP
387 for(size_t i = 0; i < outputPoints.size(); i++) {
388
389 // skip computation if i in filtered
390 if(mask != nullptr && mask[i] == 0) {
391 tmpStorage[i] = outputPoints[i];
392 continue;
393 }
394 tmpStorage[i] = this->relax(i, outputPoints, triangulationToSmooth);
395
396 VisitedMask vm{trianglesTested, visitedTriangles};
397
398 // replace curr in outputPoints_ by its projection
399 const auto res
400 = this->findProjection(ProjectionInput{tmpStorage[i], nearestVertexId[i]},
401 vm, dists, trianglesToTest, triangulationSurface);
402
403 tmpStorage[i] = res.pt;
404 nearestVertexId[i] = res.nearestVertex;
405 }
406
407 std::swap(outputPoints, tmpStorage);
408
409 this->printMsg("Projected " + std::to_string(outputPoints.size()) + " points",
410 1.0, tm.getElapsedTime(), this->threadNumber_,
411 debug::LineMode::NEW, debug::Priority::DETAIL);
412
413 return 0;
414}

416template <typename triangulationType0, typename triangulationType1>
417int ttk::SurfaceGeometrySmoother::execute(
418 float *const outputCoords,
419 const float *const inputCoords,
420 const char *const mask,
421 const SimplexId *const vertsId,
422 const int nIter,
423 const triangulationType0 &triangulationToSmooth,
424 const triangulationType1 &triangulationSurface) const {
425
426 const auto nPoints{triangulationToSmooth.getNumberOfVertices()};
427 if(triangulationSurface.getDimensionality() != 2) {
428 this->printErr("Can only project onto a surface");
429 return -1;
430 }
431
432 if(triangulationToSmooth.getDimensionality() < 1
433 || triangulationToSmooth.getDimensionality() > 2) {
434 this->printErr("Can only project a 1D or a 2D triangulated object");
435 return -1;
436 }
437
438 Timer tm{};
439 this->printMsg("Smoothing " + std::to_string(nPoints) + " in "
440 + std::to_string(nIter) + " iterations...");
441
442 // list of triangle IDs already tested
443 // (takes more memory to reduce computation time)
444 std::vector<bool> trianglesTested(
445 triangulationSurface.getNumberOfTriangles(), false);
446 std::vector<SimplexId> visitedTriangles{};
447 // distance between every mesh point and current point
448 std::vector<float> dists(triangulationSurface.getNumberOfVertices());
449
450 // temporary storage
451 std::vector<ttk::SurfaceGeometrySmoother::Point> outputPoints(nPoints),
452 tmpStorage(nPoints);
453 std::vector<SimplexId> nearestVertexId(nPoints);
454
455 // copy input
456#ifdef TTK_ENABLE_OPENMP
457#pragma omp parallel for num_threads(threadNumber_)
458#endif // TTK_ENABLE_OPENMP
459 for(SimplexId i = 0; i < nPoints; ++i) {
460 outputPoints[i][0] = inputCoords[3 * i + 0];
461 outputPoints[i][1] = inputCoords[3 * i + 1];
462 outputPoints[i][2] = inputCoords[3 * i + 2];
463 }
464
465 // ttkVertexScalarField is optional (helps for instance with
466 // MorseSmaleComplex 1-separatrices)
467 if(vertsId != nullptr) {
468#ifdef TTK_ENABLE_OPENMP
469#pragma omp parallel for num_threads(threadNumber_)
470#endif // TTK_ENABLE_OPENMP
471 for(SimplexId i = 0; i < nPoints; ++i) {
472 nearestVertexId[i] = vertsId[i];
473 }
474 } else {
475 // generate a ttkVertexScalarField-like point data array using raw
476 // euclidian distance between the points to smooth and every
477 // vertex of the surface
478 Timer tm_nv{};
479 this->printMsg("Computing nearest vertices...", debug::Priority::INFO,
480 debug::LineMode::REPLACE);
481#ifdef TTK_ENABLE_OPENMP
482#pragma omp parallel for num_threads(threadNumber_) firstprivate(dists)
483#endif // TTK_ENABLE_OPENMP
484 for(SimplexId i = 0; i < nPoints; ++i) {
485 nearestVertexId[i] = this->getNearestSurfaceVertex(
486 outputPoints[i], dists, triangulationSurface);
487 }
488 this->printMsg("Computed nearest vertices", 1.0, tm_nv.getElapsedTime(),
489 this->threadNumber_);
490 }
491
492 for(int i = 0; i < nIter; ++i) {
493 this->relaxProject(outputPoints, tmpStorage, nearestVertexId,
494 trianglesTested, visitedTriangles, dists, mask,
495 triangulationToSmooth, triangulationSurface);
496 }
497
498 // copy output
499#ifdef TTK_ENABLE_OPENMP
500#pragma omp parallel for num_threads(threadNumber_)
501#endif // TTK_ENABLE_OPENMP
502 for(SimplexId i = 0; i < nPoints; ++i) {
503 outputCoords[3 * i + 0] = outputPoints[i][0];
504 outputCoords[3 * i + 1] = outputPoints[i][1];
505 outputCoords[3 * i + 2] = outputPoints[i][2];
506 }
507
508 this->printMsg("Smoothed " + std::to_string(nPoints) + " points", 1.0,
509 tm.getElapsedTime(), this->threadNumber_);
510
511 return 0;
512}
