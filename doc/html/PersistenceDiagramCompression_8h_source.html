277 int ttk::TopologicalCompression::PerformSimplification(
278  const std::vector<std::tuple<int, double, int>> &constraints,
279  int nbConstraints,
280  int vertexNumber,
281  double *array)
282 {
283  std::vector<int> inputOffsets(vertexNumber);
284  std::vector<int> critConstraints(nbConstraints);
285  std::vector<double> inArray(vertexNumber);
286  // std::vector<int> *oo = new std::vector<int>(vertexNumber);
287  decompressedOffsets_.resize(vertexNumber); // oo->data();
288  int status = 0;
289 
290  // Offsets
291  for (int i = 0; i < vertexNumber; ++i)
292  inputOffsets[i] = i;
293 
294  // Triangulate.
295  topologicalSimplification.setupTriangulation(triangulation_);
296 
297  // Preprocess simplification.
298  // std::vector<int>* authorizedSaddles = new std::vector<int>();
299  for (int i = 0; i < nbConstraints; ++i) {
300  std::tuple<int, double, int> t = constraints[i];
301  int id = std::get<0>(t);
302  double val = std::get<1>(t);
303  int type = std::get<2>(t);
304 
305  // if (type == 0)
306  // authorizedSaddles->push_back(id);
307 
308  array[id] = val;
309 
310  // Smoothe neighborhood (along with offsets).
311  SimplexId neighborNumber = triangulation_->getVertexNeighborNumber(id);
312  for(SimplexId j = 0; j < neighborNumber; ++j) {
313  SimplexId neighbor;
314  triangulation_->getVertexNeighbor(id, j, neighbor);
315 
316  if (type == 1) { // Local_maximum.
317  if (array[neighbor] > val) array[neighbor] = val;
318  if (array[neighbor] == val && inputOffsets[neighbor] > inputOffsets[id]) {
319  int tmp = inputOffsets[id];
320  inputOffsets[id] = inputOffsets[neighbor];
321  inputOffsets[neighbor] = tmp;
322  }
323  } else if (type == -1) { // Local_minimum.
324  if (array[neighbor] < val) array[neighbor] = val;
325  if (array[neighbor] == val && inputOffsets[neighbor] < inputOffsets[id]) {
326  int tmp = inputOffsets[id];
327  inputOffsets[id] = inputOffsets[neighbor];
328  inputOffsets[neighbor] = tmp;
329  }
330  } else if (type == 0) { // Saddle
331  }
332  }
333 
334  critConstraints[i] = id;
335  }
336 
337  for (int i = 0; i < vertexNumber; ++i) inArray[i] = array[i];
338  for (int i = 0; i < vertexNumber; ++i) decompressedOffsets_[i] = 0;
339 
340  topologicalSimplification.setInputScalarFieldPointer(inArray.data());
341  topologicalSimplification.setOutputScalarFieldPointer(array);
342  topologicalSimplification.setInputOffsetScalarFieldPointer(inputOffsets.data());
343  topologicalSimplification.setOutputOffsetScalarFieldPointer(decompressedOffsets_.data());
344  topologicalSimplification.setVertexIdentifierScalarFieldPointer(critConstraints.data());
345  topologicalSimplification.setConstraintNumber(nbConstraints);

352 void ttk::TopologicalCompression::CropIntervals(
353  std::vector<std::tuple<dataType, int>> &mappings,
354  std::vector<std::tuple<dataType, int>> &mappingsSortedPerValue,
355  double min,
356  double max,

358  double *array,
359  std::vector<int>& segmentation)

361  int numberOfMisses = 0;
362  for (int i = 0; i < vertexNumber; ++i)
363  {
364  int seg = segmentation[i];
365  auto end = mappings.end();
366  auto it = lower_bound(mappings.begin(), mappings.end(), std::make_tuple(0, seg), cmp);
367  if (it != end) {
368  std::tuple<dataType, int> tt = *it;
369  double value = std::get<0>(tt);
370  int sseg = std::get<1>(tt);
371  if (seg != sseg)

376  d.dMsg(std::cout, msg.str(), ttk::Debug::infoMsg);

378

380  mappingsSortedPerValue.end(),
381  std::make_tuple(value, 0), cmp2);
382 
383  if (it2 != mappingsSortedPerValue.end()
384  && it2 != mappingsSortedPerValue.begin()
385  && (it2+1) != mappingsSortedPerValue.end())
386  {
387  dataType vv0 = std::get<0>(*(it2 - 1));
388  // double vv1 = std::get<0>(*(it2));
389  dataType vv2 = std::get<0>(*(++it2));
390  // double m0 = (vv0 + vv1) / 2;
391  // double m1 = (vv1 + vv2) / 2;
392 
393  if (array[i] < vv0) {
394  numberOfMisses++;
395  array[i] = vv0;
396  } else if (array[i] > vv2) {
397  numberOfMisses++;
398  array[i] = vv2;
399  }
400  } else {
401  if (
402  (it2 == mappingsSortedPerValue.end() || (it2+1) == mappingsSortedPerValue.end())
403  && array[i] > max)
404  {
405  numberOfMisses++;
406  array[i] = max;
407  } else if (
408  (it2 == mappingsSortedPerValue.begin() || (it2-1) == mappingsSortedPerValue.begin())
409  && array[i] < min)
410  {
411  numberOfMisses++;
412  array[i] = min;
413  }
414  }
415  } else {
416  ttk::Debug d;
417  std::stringstream msg;
418  msg << "[TopologicalCompression] Error looking for topo index."
419  << std::endl;

421  }
422  }
423 
424  if (numberOfMisses > 0) {
