325 std::pair<ttk::SimplexId, ttk::SimplexId>
326  ttk::ScalarFieldCriticalPoints::getNumberOfLowerUpperComponents(
327  const SimplexId vertexId,
328  const SimplexId *const offsets,
329  const triangulationType *triangulation) const {
330 
331  SimplexId neighborNumber = triangulation->getVertexNeighborNumber(vertexId);
332  std::vector<SimplexId> lowerNeighbors, upperNeighbors;
333 
334  for(SimplexId i = 0; i < neighborNumber; i++) {
335  SimplexId neighborId = 0;
336  triangulation->getVertexNeighbor(vertexId, i, neighborId);
337 
338  if(offsets[neighborId] < offsets[vertexId]) {
339  lowerNeighbors.push_back(neighborId);
340  }
341 
342  // upper link
343  if(offsets[neighborId] > offsets[vertexId]) {
344  upperNeighbors.push_back(neighborId);
345  }
346  }
347 
348  // shortcut, if min or max do not construct the complete star
349  if(!forceNonManifoldCheck && lowerNeighbors.empty()) {
350  // minimum
351  return std::make_pair(0, 1);
352  }
353 
354  if(!forceNonManifoldCheck && upperNeighbors.empty()) {
355  // maximum
356  return std::make_pair(1, 0);
357  }
358 
359  // now do the actual work
360  std::vector<UnionFind> lowerSeeds(lowerNeighbors.size());
361  std::vector<UnionFind *> lowerList(lowerNeighbors.size());
362  std::vector<UnionFind> upperSeeds(upperNeighbors.size());
363  std::vector<UnionFind *> upperList(upperNeighbors.size());
364 
365  for(SimplexId i = 0; i < (SimplexId)lowerSeeds.size(); i++) {
366  lowerList[i] = &(lowerSeeds[i]);
367  }
368  for(SimplexId i = 0; i < (SimplexId)upperSeeds.size(); i++) {
369  upperList[i] = &(upperSeeds[i]);
370  }
371 
372  SimplexId vertexStarSize = triangulation->getVertexStarNumber(vertexId);
373 
374  for(SimplexId i = 0; i < vertexStarSize; i++) {
375  SimplexId cellId = 0;
376  triangulation->getVertexStar(vertexId, i, cellId);
377 
378  SimplexId cellSize = triangulation->getCellVertexNumber(cellId);
379  for(SimplexId j = 0; j < cellSize; j++) {
380  SimplexId neighborId0 = -1;
381  triangulation->getCellVertex(cellId, j, neighborId0);
382 
383  if(neighborId0 != vertexId) {
384  // we are on the link
385 
386  bool lower0 = offsets[neighborId0] < offsets[vertexId];
387 
388  // connect it to everybody except himself and vertexId
389  for(SimplexId k = j + 1; k < cellSize; k++) {
390 
391  SimplexId neighborId1 = -1;
392  triangulation->getCellVertex(cellId, k, neighborId1);
393 
394  if((neighborId1 != neighborId0) && (neighborId1 != vertexId)) {
395 
396  bool lower1 = offsets[neighborId1] < offsets[vertexId];
397 
398  std::vector<SimplexId> *neighbors = &lowerNeighbors;
399  std::vector<UnionFind *> *seeds = &lowerList;
400 
401  if(!lower0) {
402  neighbors = &upperNeighbors;
403  seeds = &upperList;
404  }
405 
406  if(lower0 == lower1) {
407  // connect their union-find sets!
408  SimplexId lowerId0 = -1, lowerId1 = -1;
409  for(SimplexId l = 0; l < (SimplexId)neighbors->size(); l++) {
410  if((*neighbors)[l] == neighborId0) {
411  lowerId0 = l;
412  }
413  if((*neighbors)[l] == neighborId1) {
414  lowerId1 = l;
415  }
416  }
417  if((lowerId0 != -1) && (lowerId1 != -1)) {
418  (*seeds)[lowerId0] = UnionFind::makeUnion(
419  (*seeds)[lowerId0], (*seeds)[lowerId1]);
420  (*seeds)[lowerId1] = (*seeds)[lowerId0];
421  }
422  }
423  }
424  }
425  }
426  }
427  }
428 
429  // let's remove duplicates now
430 
431  // update the UF if necessary
432  for(SimplexId i = 0; i < (SimplexId)lowerList.size(); i++)
433  lowerList[i] = lowerList[i]->find();
434  for(SimplexId i = 0; i < (SimplexId)upperList.size(); i++)
435  upperList[i] = upperList[i]->find();
436 
437  std::vector<UnionFind *>::iterator it;
438  std::sort(lowerList.begin(), lowerList.end());
439  it = unique(lowerList.begin(), lowerList.end());
440  lowerList.resize(distance(lowerList.begin(), it));
441 
442  std::sort(upperList.begin(), upperList.end());
443  it = unique(upperList.begin(), upperList.end());
444  upperList.resize(distance(upperList.begin(), it));
445 
446  if(debugLevel_ >= (int)(debug::Priority::VERBOSE)) {
447  printMsg("Vertex #" + std::to_string(vertexId)
448  + ": lowerLink-#CC=" + std::to_string(lowerList.size())
449  + " upperLink-#CC=" + std::to_string(upperList.size()),
450  debug::Priority::VERBOSE);
451  }
452 
453  return std::make_pair(lowerList.size(), upperList.size());
454 }

456 template <class triangulationType>
457 char ttk::ScalarFieldCriticalPoints::getCriticalType(
458  const SimplexId &vertexId,
459  const SimplexId *const offsets,
460  const triangulationType *triangulation) const {
461 
462  SimplexId downValence, upValence;
463  std::tie(downValence, upValence)
464  = getNumberOfLowerUpperComponents(vertexId, offsets, triangulation);
465 
466  if(downValence == 0 && upValence == 1) {
467  return (char)(CriticalType::Local_minimum);
468  } else if(downValence == 1 && upValence == 0) {
469  return (char)(CriticalType::Local_maximum);
470  } else if(downValence == 1 && upValence == 1) {
471  // regular point
472  return (char)(CriticalType::Regular);
473  } else {
474  // saddles
475  if(dimension_ == 2) {
476  if((downValence == 2 && upValence == 1)
477  || (downValence == 1 && upValence == 2)
478  || (downValence == 2 && upValence == 2)) {
479  // regular saddle
480  return (char)(CriticalType::Saddle1);
481  } else {
482  // monkey saddle, saddle + extremum
483  return (char)(CriticalType::Degenerate);
484  // NOTE: you may have multi-saddles on the boundary in that
485  // configuration
486  // to make this computation 100% correct, one would need to
487  // disambiguate boundary from interior vertices
488  }
489  } else if(dimension_ == 3) {
490  if(downValence == 2 && upValence == 1) {
491  return (char)(CriticalType::Saddle1);
492  } else if(downValence == 1 && upValence == 2) {
493  return (char)(CriticalType::Saddle2);
494  } else {
495  // monkey saddle, saddle + extremum
496  return (char)(CriticalType::Degenerate);
497  // NOTE: we may have a similar effect in 3D (TODO)
498  }
499  }
500  }
501 
502  // -2: regular points
503  return (char)(CriticalType::Regular);
504 }
