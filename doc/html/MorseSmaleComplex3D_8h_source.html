893  sepsByThread[i].emplace_back(s1, s2);
894  }
895  }

899  std::vector<size_t> partialSepsId(sepsByThread.size() + 1, 0);

907  separatricesGeometry.resize(partialSepsId.back());

910 #pragma omp parallel for num_threads(threadNumber_)
911 #endif // TTK_ENABLE_OPENMP

913  for(size_t j = 0; j < sepsByThread[i].size(); ++j) {
914  const auto &sads = sepsByThread[i][j];
915  const size_t k = partialSepsId[i] + j;
916  separatrices[k] = Separatrix{
917  true, sads.first, sads.second, false, static_cast<SimplexId>(k)};
918  separatricesGeometry[k] = std::move(sepsGeomByThread[i][j]);
