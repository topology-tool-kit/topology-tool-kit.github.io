665  template <class dataType>
666  void branchDecompositionToTree(ftm::FTMTree_MT *tree) {
667  ftm::idNode treeRoot = tree->getRoot();
668 
669  // One pair case
670  if(tree->isThereOnlyOnePersistencePair()) {
671  idNode treeRootOrigin = tree->getNode(treeRoot)->getOrigin();
672  tree->getNode(treeRootOrigin)->setOrigin(treeRoot);
673  return;
674  }
675 
676  // Manage full merge and dontuseMinMaxPair_
677  bool isFM = tree->isFullMerge();
678  if(isFM) {
679  ftm::idNode mergedRootOrigin = getMergedRootOrigin<dataType>(tree);
680  if(not tree->isNodeIdInconsistent(mergedRootOrigin)
681  and mergedRootOrigin != treeRoot)
682  tree->getNode(treeRoot)->setOrigin(mergedRootOrigin);
683  else {
684  printErr("branchDecompositionToTree mergedRootOrigin inconsistent");
685  }
686  }
687 
688  // Some functions
689  bool isJT = tree->isJoinTree<dataType>();
690  auto comp = [&](const std::tuple<ftm::idNode, dataType> &a,
691  const std::tuple<ftm::idNode, dataType> &b) {
692  return isJT ? std::get<1>(a) > std::get<1>(b)
693  : std::get<1>(a) < std::get<1>(b);
694  };
695  auto getIndexNotMultiPers = [&](int index, ftm::FTMTree_MT *treeT,
696  std::vector<ftm::idNode> &children) {
697  while(treeT->isMultiPersPair(children[index]))
698  --index;
699  index = std::max(0, index);
700  return index;
701  };
702 
703  // Branch Decomposition To Tree
704  std::vector<std::tuple<ftm::idNode, ftm::idNode>> nodeParent;
705  std::queue<ftm::idNode> queue;
706  queue.emplace(treeRoot);
707  while(!queue.empty()) {
708  ftm::idNode node = queue.front();
709  queue.pop();
710  auto nodeOrigin = tree->getNode(node)->getOrigin();
711  if(tree->isLeaf(node)) {
712  if(tree->isNodeAlone(nodeOrigin)) {
713  if(not isFM)
714  nodeParent.push_back(std::make_tuple(nodeOrigin, node));
715  else
716  nodeParent.push_back(std::make_tuple(node, nodeOrigin));
717  } else if(tree->isMultiPersPair(node)) {
718  nodeParent.push_back(std::make_tuple(node, nodeOrigin));
719  }
720  continue;
721  }
722 
723  // Get children and sort them by scalar values
724  std::vector<ftm::idNode> childrenOri;
725  tree->getChildren(node, childrenOri);
726  std::vector<ftm::idNode> children = childrenOri;
727  std::vector<std::tuple<ftm::idNode, dataType>> childrenScalars;
728  for(unsigned int i = 0; i < children.size(); ++i) {
729  if(isFM and (int) children[i] != nodeOrigin)
730  children[i] = tree->getNode(children[i])->getOrigin();
731  childrenScalars.push_back(
732  std::make_tuple(children[i], tree->getValue<dataType>(children[i])));
733  }
734  std::sort(std::begin(childrenScalars), std::end(childrenScalars), comp);
735  children.clear();
736  for(unsigned int i = 0; i < childrenScalars.size(); ++i)
737  children.push_back(std::get<0>(childrenScalars[i]));
738 
739  // Get new parent of children
740  for(unsigned int i = 1; i < children.size(); ++i) {
741  if(tree->isMultiPersPair(children[i]))
742  continue;
743  int index = getIndexNotMultiPers(i - 1, tree, children);
744  nodeParent.push_back(std::make_tuple(children[i], children[index]));
745  }
746 
747  bool multiPersPair = tree->getNode(nodeOrigin)->getOrigin() != (int)node;
748  if(not multiPersPair) {
749  if(not isFM) {
750  int index = getIndexNotMultiPers(children.size() - 1, tree, children);
751  nodeParent.push_back(std::make_tuple(nodeOrigin, children[index]));
752  } else
753  nodeParent.push_back(std::make_tuple(children[0], node));
754  } else {
755  // std::cout << "branchDecompositionToTree multiPersPair" << std::endl;
756  nodeParent.push_back(std::make_tuple(children[0], nodeOrigin));
757  int index = getIndexNotMultiPers(children.size() - 1, tree, children);
758  nodeParent.push_back(std::make_tuple(node, children[index]));
759  }
760 
761  // Push children to the queue
762  for(auto child : childrenOri)
763  queue.emplace(child);
764  }
765 
766  // Set new parents for each node
767  for(auto nodeParentT : nodeParent)
768  tree->setParent(std::get<0>(nodeParentT), std::get<1>(nodeParentT));
769 
770  // Verify that the tree is correct
771  for(unsigned int i = 0; i < tree->getNumberOfNodes(); ++i)
772  if(tree->getNode(i)->getNumberOfDownSuperArcs() == 1
773  and tree->getNode(i)->getNumberOfUpSuperArcs() == 1) {
774  tree->printTree();
775  std::stringstream ss;
776  ss << i << " _ " << tree->getNode(i)->getOrigin();
777  printMsg(ss.str());
778  printErr("1 up arc and 1 down arc");
779  }
