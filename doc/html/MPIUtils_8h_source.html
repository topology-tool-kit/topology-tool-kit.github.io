95 template <typename T>
96 int gatherVectors(std::vector<std::vector<T>> &dst,
97 const std::vector<T> &src,
98 const int destRank) {
99
100 if(!ttk::isRunningWithMPI()) {
101 return -1;
102 }
103
104 // src sizes (gathered on rank 0)
105 std::vector<unsigned long> vecSizes{};
106
107 if(ttk::MPIrank_ == destRank) {
108 vecSizes.resize(MPIsize_);
109 dst.resize(MPIsize_);
110 }
111
112 const unsigned long localSize = src.size();
113 // gather src sizes on destRank
114 MPI_Gather(&localSize, 1, MPI_UNSIGNED_LONG, vecSizes.data(), 1,
115 MPI_UNSIGNED_LONG, destRank, ttk::MPIcomm_);
116
117 if(ttk::MPIrank_ == destRank) {
118 // allocate dst with vecSizes
119 for(int i = 0; i < ttk::MPIsize_; ++i) {
120 if(i == destRank) {
121 continue;
122 }
123 dst[i].resize(vecSizes[i]);
124 }
125
126 for(int i = 0; i < ttk::MPIsize_; ++i) {
127 if(i == destRank) {
128 continue;
129 }
130 // receive src content from other ranks
131 MPI_Recv(dst[i].data(), dst[i].size(), ttk::getMPIType(src[0]), i,
132 MPI_ANY_TAG, ttk::MPIcomm_, MPI_STATUS_IGNORE);
133 }
134 dst[destRank] = std::move(src);
135
136 } else {
137 // send src content to destRank
138 MPI_Send(src.data(), src.size(), ttk::getMPIType(src[0]), destRank, 0,
139 ttk::MPIcomm_);
140 }
141
142 return 0;
143 }

163 template <typename DT, typename IT>
164 int getGhostCellScalars(DT *scalarArray,
165 const int *const rankArray,
166 const IT *const globalIds,
167 const std::unordered_map<IT, IT> &gidToLidMap,
168 const std::unordered_set<int> &neighbors,
169 const int rankToSend,
170 const IT nVerts,
171 MPI_Comm communicator) {
172 if(!ttk::isRunningWithMPI()) {
173 return -1;
174 }
175 MPI_Datatype MPI_DT = getMPIType(static_cast<DT>(0));
176 MPI_Datatype MPI_IT = getMPIType(static_cast<IT>(0));
177 // we need unique tags for each rankToSend, otherwise messages might become
178 // entangled
179 int tagMultiplier = rankToSend + 1;
180 int amountTag = 101 * tagMultiplier;
181 int idsTag = 102 * tagMultiplier;
182 int valuesTag = 103 * tagMultiplier;
183 if(rankToSend == ttk::MPIrank_) {
184 // initialize the inner vectors with size 0
185 std::vector<std::vector<IT>> rankVectors(
186 ttk::MPIsize_, std::vector<IT>(0));
187 // aggregate the needed ids
188
189 for(IT i = 0; i < nVerts; i++) {
190 if(ttk::MPIrank_ != rankArray[i]) {
191 rankVectors[rankArray[i]].push_back(globalIds[i]);
192 }
193 }
194 // send the amount of ids and the needed ids themselves
195 for(int r = 0; r < ttk::MPIsize_; r++) {
196 if(ttk::MPIrank_ != r && neighbors.find(r) != neighbors.end()) {
197 IT nValues = rankVectors[r].size();
198 MPI_Send(&nValues, 1, MPI_IT, r, amountTag, communicator);
199 if(nValues > 0) {
200 MPI_Send(
201 rankVectors[r].data(), nValues, MPI_IT, r, idsTag, communicator);
202 }
203 }
204 }
205 // receive the scalar values
206 for(int r = 0; r < ttk::MPIsize_; r++) {
207 if(ttk::MPIrank_ != r && neighbors.find(r) != neighbors.end()) {
208 IT nValues = rankVectors[r].size();
209 std::vector<DT> receivedValues(nValues);
210 if(nValues > 0) {
211 MPI_Recv(receivedValues.data(), nValues, MPI_DT, r, valuesTag,
212 communicator, MPI_STATUS_IGNORE);
213 for(IT i = 0; i < nValues; i++) {
214 DT receivedVal = receivedValues[i];
215 IT globalId = rankVectors[r][i];
216 IT localId = gidToLidMap.at(globalId);
217 scalarArray[localId] = receivedVal;
218 }
219 }
220 }
221 }
222
223 } else { // owner ranks
224 // if rankToSend is not the neighbor of the current rank, we do not need
225 // to do anything
226 if(neighbors.find(rankToSend) != neighbors.end()) {
227 // receive the amount of ids and the needed ids themselves
228 IT nValues;
229
230 MPI_Recv(&nValues, 1, MPI_IT, rankToSend, amountTag, communicator,
231 MPI_STATUS_IGNORE);
232
233 if(nValues > 0) {
234 std::vector<IT> receivedIds(nValues);
235 MPI_Recv(receivedIds.data(), nValues, MPI_IT, rankToSend, idsTag,
236 communicator, MPI_STATUS_IGNORE);
237
238 // assemble the scalar values
239 std::vector<DT> valuesToSend(nValues);
240 for(IT i = 0; i < nValues; i++) {
241 IT globalId = receivedIds[i];
242 IT localId = gidToLidMap.at(globalId);
243 valuesToSend[i] = scalarArray[localId];
244 }
245
246 // send the scalar values
247 MPI_Send(valuesToSend.data(), nValues, MPI_DT, rankToSend, valuesTag,
248 communicator);
249 }
250 }
251 }
252
253 return 0;
254 }

265 template <typename IT>
266 int getNeighbors(std::unordered_set<int> &neighbors,
267 const int *const rankArray,
268 const IT nVerts,
269 MPI_Comm communicator) {
270 for(IT i = 0; i < nVerts; i++) {
271 if(rankArray[i] != ttk::MPIrank_) {
272 neighbors.emplace(rankArray[i]);
273 }
274 }
275 std::vector<int> sendVector(neighbors.begin(), neighbors.end());
276 int localSize = neighbors.size();
277 int sizes[ttk::MPIsize_];
278 int displacements[ttk::MPIsize_];
279 MPI_Gather(&localSize, 1, MPI_INT, sizes, 1, MPI_INT, 0, communicator);
280 int totalSize = 0;
281 if(ttk::MPIrank_ == 0) {
282 for(int i = 0; i < ttk::MPIsize_; i++) {
283 totalSize += sizes[i];
284 if(i == 0) {
285 displacements[i] = 0;
286 } else {
287 displacements[i] = displacements[i - 1] + sizes[i - 1];
288 }
289 }
290 }
291 std::vector<int> rootVector(totalSize);
292
293 MPI_Gatherv(sendVector.data(), sendVector.size(), MPI_INT,
294 rootVector.data(), sizes, displacements, MPI_INT, 0,
295 communicator);
296 std::vector<int> scatterVector;
297
298 if(ttk::MPIrank_ == 0) {
299 // now we transform this 1d vector in a correct vector of sets which we
300 // will transform back to scatter it
301 std::vector<std::unordered_set<int>> setsFromRanks(ttk::MPIsize_);
302 auto begin = rootVector.begin();
303 auto end = rootVector.begin();
304 for(int i = 0; i < ttk::MPIsize_; i++) {
305 end = begin + sizes[i];
306 std::unordered_set<int> s(begin, end);
307 setsFromRanks[i] = s;
308 begin = end;
309 // std::cout << "R" << std::to_string(i) << " nr needs something from "
310 // << std::to_string(setsFromRanks[i].size()) << " neighbors." <<
311 // std::endl;
312 }
313 // now we need to check for each rank if they are a neighbor of any other
314 // rank. If so, we need to add those to the neighbors
315 for(int i = 0; i < ttk::MPIsize_; i++) {
316 for(int j = 0; i < ttk::MPIsize_; i++) {
317 if(setsFromRanks[j].find(i) != setsFromRanks[j].end()) {
318 setsFromRanks[i].emplace(j);
319 }
320 }
321 }
322 // now we transform this vector of sets back into a 1d vector to scatter
323 // it
324 for(int i = 0; i < ttk::MPIsize_; i++) {
325 sizes[i] = setsFromRanks[i].size();
326 if(i == 0) {
327 displacements[i] = 0;
328 } else {
329 displacements[i] = displacements[i - 1] + sizes[i - 1];
330 }
331 scatterVector.insert(scatterVector.end(), setsFromRanks[i].begin(),
332 setsFromRanks[i].end());
333 }
334 }
335
336 // scatter first the size and then the vector itself
337 int receivedSize;
338 MPI_Scatter(sizes, 1, MPI_INT, &receivedSize, 1, MPI_INT, 0, communicator);
339
340 // and then the actual neighbors
341 std::vector<int> receivedNeighbors(receivedSize);
342 MPI_Scatterv(scatterVector.data(), sizes, displacements, MPI_INT,
343 receivedNeighbors.data(), receivedSize, MPI_INT, 0,
344 communicator);
345 // then we turn the vector back into a set
346 std::unordered_set<int> finalSet(
347 receivedNeighbors.begin(), receivedNeighbors.end());
348 neighbors = finalSet;
349
350 return 0;
351 }

389 // returns true if bounding boxes intersect, false if not
390 bool inline checkForIntersection(double *myBB, double *theirBB) {
391 return !(
392 myBB[0] > theirBB[1] // my left side is right of their right side
393 || myBB[1] < theirBB[0] // my right side is left of their left side
394 || myBB[2] > theirBB[3] // my bottom side is above their top side
395 || myBB[3] < theirBB[2] // my top side is under their bottom side
396 || myBB[4] > theirBB[5] // my front side is behind their back side
397 || myBB[5] < theirBB[4] // my back side is in front of their front side
398 );
399 }

409 void inline produceRankArray(std::vector<int> &rankArray,
410 long int *globalIds,
411 unsigned char *ghostCells,
412 int nVertices,
413 double *boundingBox) {
414 std::vector<std::array<double, 6>> rankBoundingBoxes(
415 ttk::MPIsize_, std::array<double, 6>({}));
416 std::copy(
417 boundingBox, boundingBox + 6, rankBoundingBoxes[ttk::MPIrank_].begin());
418 for(int r = 0; r < ttk::MPIsize_; r++) {
419 MPI_Bcast(rankBoundingBoxes[r].data(), 6, MPI_DOUBLE, r, ttk::MPIcomm_);
420 }
421
422 double epsilon = 0.00001;
423 // inflate our own bounding box by epsilon
424 for(int i = 0; i < 6; i++) {
425 if(i % 2 == 0)
426 boundingBox[i] -= epsilon;
427 if(i % 2 == 1)
428 boundingBox[i] += epsilon;
429 }
430 std::vector<int> neighbors;
431 for(int i = 0; i < ttk::MPIsize_; i++) {
432 if(i != ttk::MPIrank_) {
433 double *theirBoundingBox = rankBoundingBoxes[i].data();
434 if(checkForIntersection(boundingBox, theirBoundingBox)) {
435 neighbors.push_back(i);
436 }
437 }
438 }
439 MPI_Datatype MIT = ttk::getMPIType(static_cast<ttk::SimplexId>(0));
440 std::vector<ttk::SimplexId> currentRankUnknownIds;
441 std::vector<std::vector<ttk::SimplexId>> allUnknownIds(ttk::MPIsize_);
442 std::unordered_set<ttk::SimplexId> gIdSet;
443 std::unordered_map<ttk::SimplexId, ttk::SimplexId> gIdToLocalMap;
444
445 for(int i = 0; i < nVertices; i++) {
446 int ghostCellVal = ghostCells[i];
447 ttk::SimplexId globalId = globalIds[i];
448 if(ghostCellVal == 0) {
449 // if the ghost cell value is 0, then this vertex mainly belongs to
450 // this rank
451 rankArray[i] = ttk::MPIrank_;
452 gIdSet.insert(globalId);
453 } else {
454 // otherwise the vertex belongs to another rank and we need to find
455 // out to which one this needs to be done by broadcasting the global
456 // id and hoping for some other rank to answer
457 currentRankUnknownIds.push_back(globalId);
458 gIdToLocalMap[globalId] = i;
459 }
460 }
461
462 ttk::SimplexId sizeOfCurrentRank = currentRankUnknownIds.size();
463 std::vector<ttk::SimplexId> gIdsToSend;
464 std::vector<ttk::SimplexId> receivedGlobals;
465 receivedGlobals.resize(sizeOfCurrentRank);
466 ttk::SimplexId sizeOfNeighbor;
467 std::vector<ttk::SimplexId> neighborUnknownIds;
468 for(int neighbor : neighbors) {
469 // we first send the size and then all needed ids to the neighbor
470 MPI_Sendrecv(&sizeOfCurrentRank, 1, MIT, neighbor, ttk::MPIrank_,
471 &sizeOfNeighbor, 1, MIT, neighbor, neighbor, ttk::MPIcomm_,
472 MPI_STATUS_IGNORE);
473 neighborUnknownIds.resize(sizeOfNeighbor);
474 gIdsToSend.reserve(sizeOfNeighbor);
475
476 MPI_Sendrecv(currentRankUnknownIds.data(), sizeOfCurrentRank, MIT,
477 neighbor, ttk::MPIrank_, neighborUnknownIds.data(),
478 sizeOfNeighbor, MIT, neighbor, neighbor, ttk::MPIcomm_,
479 MPI_STATUS_IGNORE);
480
481 // then we check if the needed globalid values are present in the local
482 // globalid set if so, we send the rank value to the requesting rank
483 for(ttk::SimplexId gId : neighborUnknownIds) {
484 if(gIdSet.count(gId)) {
485 // add the value to the vector which will be sent
486 gIdsToSend.push_back(gId);
487 }
488 }
489 MPI_Status status;
490 int amount;
491
492 MPI_Sendrecv(gIdsToSend.data(), gIdsToSend.size(), MIT, neighbor,
493 ttk::MPIrank_, receivedGlobals.data(),
494 currentRankUnknownIds.size(), MIT, neighbor, neighbor,
495 ttk::MPIcomm_, &status);
496
497 MPI_Get_count(&status, MIT, &amount);
498 receivedGlobals.resize(amount);
499
500 for(ttk::SimplexId receivedGlobal : receivedGlobals) {
501 ttk::SimplexId localVal = gIdToLocalMap[receivedGlobal];
502 rankArray[localVal] = neighbor;
503 }
504 // cleanup
505 gIdsToSend.clear();
506 receivedGlobals.resize(sizeOfCurrentRank);
507 }
508 }

522

583 template <typename DT, typename IT>
584 void getMax(int intTag,
585 int structTag,
586 IT &currentOrder,
587 int burstSize,
588 MPI_Datatype MPI_IT,
589 std::vector<value<DT, IT>> &finalValues,
590 std::vector<std::vector<value<DT, IT>>> &unsortedReceivedValues,
591 std::vector<std::vector<IT>> &orderResendValues,
592 std::vector<IT> &orderedValuesForRank,
593 std::vector<value<DT, IT>> &sortingValues) {
594 // take the current maximum scalar over all ranks
595 int rankIdOfMaxScalar = -1;
596 DT maxScalar = std::numeric_limits<DT>::lowest();
597 IT maxGId = -1;
598 for(size_t i = 0; i < unsortedReceivedValues.size(); i++) {
599 if(unsortedReceivedValues[i].size() > 0) {
600 const auto &v = unsortedReceivedValues[i].back();
601 if(v.scalar == maxScalar ? v.globalId > maxGId : v.scalar > maxScalar) {
602 maxScalar = v.scalar;
603 maxGId = v.globalId;
604 rankIdOfMaxScalar = i;
605 }
606 }
607 }
608
609 // move the struct from the unsortedReceivedValues subvector to the
610 // finalValues vector to get an ordering
611 value<DT, IT> currentValue
612 = unsortedReceivedValues[rankIdOfMaxScalar].back();
613 // we send the globalId and the the order via one send command,
614 // therefore we need to check two concurrent values at the same time
615 // later on
616 orderResendValues[rankIdOfMaxScalar].push_back(currentValue.globalId);
617 orderResendValues[rankIdOfMaxScalar].push_back(currentOrder);
618 currentOrder--;
619 finalValues.push_back(currentValue);
620 unsortedReceivedValues[rankIdOfMaxScalar].pop_back();
621 if(unsortedReceivedValues[rankIdOfMaxScalar].size() == 0) {
622 if(rankIdOfMaxScalar == 0) {
623 // append the ordered values to the correct vector
624 orderedValuesForRank.insert(
625 orderedValuesForRank.end(),
626 orderResendValues[rankIdOfMaxScalar].begin(),
627 orderResendValues[rankIdOfMaxScalar].end());
628 orderResendValues[rankIdOfMaxScalar].clear();
629
630 if(!sortingValues.empty()) {
631 std::vector<value<DT, IT>> ownValues;
632 returnVectorForBurstsize<DT, IT>(ownValues, sortingValues, burstSize);
633
634 unsortedReceivedValues[rankIdOfMaxScalar] = ownValues;
635 }
636 } else {
637 // receive more values from rank, send ordering to the rank
638 // send to the finished rank that we want more
639 MPI_Send(orderResendValues[rankIdOfMaxScalar].data(),
640 orderResendValues[rankIdOfMaxScalar].size(), MPI_IT,
641 rankIdOfMaxScalar, intTag * rankIdOfMaxScalar, ttk::MPIcomm_);
642 orderResendValues[rankIdOfMaxScalar].clear();
643 // we receive more vals, if the size is still zero afterwards, we are
644 // finished with this rank
645 ReceiveAndAddToVector(
646 rankIdOfMaxScalar, structTag, unsortedReceivedValues);
647 }
648 }

743 template <typename DT, typename IT>
744 void produceOrdering(SimplexId *orderArray,
745 const DT *scalarArray,
746 const IT *globalIds,
747 const int *rankArray,
748 const size_t nVerts,
749 const int burstSize) {
750 int intTag = 101;
751 int structTag = 102;
752
753 MPI_Barrier(ttk::MPIcomm_);
754
755 MPI_Datatype MPI_IT = ttk::getMPIType(static_cast<IT>(0));
756
757 ttk::Timer fillAndSortTimer;
758 std::vector<value<DT, IT>> sortingValues;
759 std::vector<IT> gidsToGetVector;
760 std::unordered_map<IT, IT> gidToLidMap;
761 populateVector<DT, IT>(sortingValues, gidsToGetVector, gidToLidMap, nVerts,
762 scalarArray, globalIds, rankArray);
763
764 // sort the scalar array distributed first by the scalar value itself,
765 // then by the global id
766 sortVerticesDistributed<DT, IT>(sortingValues, ttk::globalThreadNumber_);
767
768 // when all are done sorting, rank 0 requests the highest values and
769 // merges them
770 MPI_Barrier(ttk::MPIcomm_);
771
772 std::vector<IT> orderedValuesForRank;
773 std::vector<value<DT, IT>> finalValues;
774 ttk::Timer mergeTimer;
775 IT localSize = sortingValues.size();
776 IT totalSize;
777 // get the complete size of the dataset by summing up the local sizes
778 MPI_Reduce(&localSize, &totalSize, 1, MPI_IT, MPI_SUM, 0, ttk::MPIcomm_);
779 if(ttk::MPIrank_ == 0) {
780 finalValues.reserve(totalSize);
781 IT currentOrder = totalSize - 1;
782 std::vector<std::vector<value<DT, IT>>> unsortedReceivedValues;
783 unsortedReceivedValues.resize(ttk::MPIsize_);
784 std::vector<std::vector<IT>> orderResendValues;
785 orderResendValues.resize(ttk::MPIsize_);
786 for(int i = 0; i < ttk::MPIsize_; i++) {
787 orderResendValues[i].reserve(burstSize);
788 }
789
790 // receive the first batch of values
791 for(int i = 0; i < ttk::MPIsize_; i++) {
792 if(i == 0) {
793 std::vector<value<DT, IT>> ownValues;
794 returnVectorForBurstsize<DT, IT>(ownValues, sortingValues, burstSize);
795 unsortedReceivedValues[i] = ownValues;
796 } else {
797 ReceiveAndAddToVector<DT, IT>(i, structTag, unsortedReceivedValues);
798 }
799 }
800 while(finalValues.size() < (size_t)totalSize) {
801 getMax<DT, IT>(intTag, structTag, currentOrder, burstSize, MPI_IT,
802 finalValues, unsortedReceivedValues, orderResendValues,
803 orderedValuesForRank, sortingValues);
804 }
805
806 } else { // other Ranks
807 // send the next burstsize values and then wait for an answer from the
808 // root rank
809 while(!sortingValues.empty()) {
810 std::vector<value<DT, IT>> sendValues;
811 returnVectorForBurstsize<DT, IT>(sendValues, sortingValues, burstSize);
812 int size = sendValues.size();
813 MPI_Send(sendValues.data(), size * sizeof(value<DT, IT>), MPI_CHAR, 0,
814 structTag * ttk::MPIrank_, ttk::MPIcomm_);
815 std::vector<IT> receivedValues;
816
817 // be prepared to receive burstsize of elements, resize after
818 // receiving to the correct size
819 receivedValues.resize(burstSize * 2);
820 MPI_Status status;
821 int amount;
822 MPI_Recv(receivedValues.data(), burstSize * 2, MPI_IT, 0,
823 intTag * ttk::MPIrank_, ttk::MPIcomm_, &status);
824 MPI_Get_count(&status, MPI_IT, &amount);
825
826 receivedValues.resize(amount);
827 orderedValuesForRank.insert(orderedValuesForRank.end(),
828 receivedValues.begin(),
829 receivedValues.end());
830 }
831 // afterwards send once a message of length 0 to root to show that we
832 // are done
833 MPI_Send(sortingValues.data(), 0, MPI_CHAR, 0, structTag * ttk::MPIrank_,
834 ttk::MPIcomm_);
835 }
836
837 // all ranks do the following
838 MPI_Barrier(ttk::MPIcomm_);
839
840 ttk::Timer orderTimer;
841 buildArrayForReceivedData<IT>(orderedValuesForRank.size(),
842 orderedValuesForRank.data(), gidToLidMap,
843 orderArray, ttk::globalThreadNumber_);
844
845 // we receive the values at the ghostcells through the abstract
846 // exchangeGhostCells method
847 ttk::exchangeGhostCells<ttk::SimplexId, IT>(
848 orderArray, rankArray, globalIds, gidToLidMap, nVerts, ttk::MPIcomm_);
849 }
