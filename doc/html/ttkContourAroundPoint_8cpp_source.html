115bool ttkContourAroundPoint::preprocessPts(vtkUnstructuredGrid *nodes,
116 vtkUnstructuredGrid *arcs) {
117 // ---- Point data ---- //
118
119 auto points = nodes->GetPoints();
120 if(points->GetDataType() != VTK_FLOAT) {
121 printErr("The point coordinates must be of type float");
122 return false;
123 }
124 auto coords
125 = reinterpret_cast<float *>(ttkUtils::GetVoidPointer(points->GetData()));
126
127 auto pData = nodes->GetPointData();
128 auto scalarBuf = getBuffer<float>(pData, "Scalar", VTK_FLOAT, "float");
129 auto codeBuf = getBuffer<int>(pData, "CriticalType", VTK_INT, "int");
130 if(!scalarBuf || !codeBuf)
131 return false;
132
133 // ---- Cell data ---- //
134
135#ifndef NDEBUG // each arc should of course be defined by exactly two vertices
136 auto cells = arcs->GetCells();
137 const auto maxNvPerC = cells->GetMaxCellSize();
138 if(maxNvPerC != 2) {
139 printErr(
140 "The points must come in pairs but there is at least one cell with "
141 + std::to_string(maxNvPerC) + " points");
142 return false;
143 }
144 // NOTE Ideally check for minNvPerC != 2
145#endif
146
147 auto cData = arcs->GetCellData();
148 auto c2p = getBuffer<int>(cData, "upNodeId", VTK_INT, "int");
149 auto c2q = getBuffer<int>(cData, "downNodeId", VTK_INT, "int");
150 if(!c2p || !c2q)
151 return false;
152
153 // ---- Loop over pairs ---- //
154
155 static constexpr int minCode = 0;
156 static constexpr int maxCode = 3;
157 const double sadFac = ui_extension * 0.01; // saddle or mean of min and max
158 const double extFac = 1 - sadFac; // factor for the extreme point
159
160 _coords.resize(0);
161 _scalars.resize(0);
162 _isovals.resize(0);
163 _flags.resize(0);
164
165 auto addPoint = [this, coords, scalarBuf](int p, float isoval, int code) {
166 const auto point = &coords[p * 3];
167 _coords.push_back(point[0]);
168 _coords.push_back(point[1]);
169 _coords.push_back(point[2]);
170 _scalars.push_back(scalarBuf[p]);
171 _isovals.push_back(isoval);
172 _flags.push_back(code == minCode ? 0 : 1);
173 };
174
175 const vtkIdType nc = arcs->GetNumberOfCells();
176 for(vtkIdType c = 0; c < nc; ++c) {
177 const auto p = c2p[c];
178 const auto q = c2q[c];
179 const auto pCode = codeBuf[p];
180 const auto qCode = codeBuf[q];
181
182 const bool pIsSad = pCode != minCode && pCode != maxCode;
183 const bool qIsSad = qCode != minCode && qCode != maxCode;
184 if(pIsSad || qIsSad) {
185 if(pIsSad && qIsSad) // two saddles
186 continue;
187 // extremum and saddle
188 const auto ext = pIsSad ? q : p;
189 const auto sad = pIsSad ? p : q;
190 const float isoval = scalarBuf[ext] * extFac + scalarBuf[sad] * sadFac;
191 addPoint(ext, isoval, codeBuf[ext]);
192 } else { // min-max pair
193 printWrn("Arc " + std::to_string(c) + " joins a minimum and a maximum");
194 const auto pVal = scalarBuf[p];
195 const auto qVal = scalarBuf[q];
196 const auto cVal = (pVal + qVal) / 2;
197 addPoint(p, pVal * extFac + cVal * sadFac, pCode);
198 addPoint(q, qVal * extFac + cVal * sadFac, qCode);
199 }
200 }
201
202 auto np = _scalars.size();
203 const auto errorCode = this->setInputPoints(
204 _coords.data(), _scalars.data(), _isovals.data(), _flags.data(), np);
205 if(errorCode < 0) {
206 printErr("setInputPoints failed with code " + std::to_string(errorCode));
207 return false;
208 }
209 return true;
210}
