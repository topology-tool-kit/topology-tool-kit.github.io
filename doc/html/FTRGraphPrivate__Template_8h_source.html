874  template <typename ScalarType>
875  bool FTRGraph<ScalarType>::checkLast(Propagation *const localProp,
876  const std::vector<idEdge> &starVect) {
877  const idVertex curSaddle = localProp->getCurVertex();
878  AtomicUF *curId = localProp->getId();
879  valence decr = 0;
880 
881  // NOTE:
882  // Using propagation id allows to decrement by the number of time this
883  // propagation has reached the saddle, even if the propagation take care
884  // of several of these arcs (after a Hole-split).
885  for(idEdge edgeId : starVect) {
886  const idSuperArc edgeArc = dynGraph(localProp).getSubtreeArc(edgeId);
887  if(edgeArc == nullSuperArc) {
888  continue;
889  }
890  AtomicUF *tmpId = graph_.getArc(edgeArc).getPropagation()->getId();
891  if(tmpId == curId) {
892  graph_.getArc(edgeArc).setEnd(curSaddle);
893  ++decr;
894  }
895  }
896 
897 #if defined(__GNUC__)
898 #pragma GCC diagnostic push
899 #pragma GCC diagnostic ignored "-Wunused-value"
900 #endif // __GNUC__
901 
902  valence oldVal = 0;
903  if(localProp->goUp()) {
904  // for gcc 4.8 and old openMP
905  valence *const vd = &graph_.valDown_[curSaddle];
906 #ifdef TTK_ENABLE_OPENMP
907 #pragma omp atomic capture
908 #endif
909  {
910  oldVal = *vd;
911  *vd -= decr;
912  }
913 
914  } else {
915  valence *const vu = &graph_.valUp_[curSaddle];
916 #ifdef TTK_ENABLE_OPENMP
917 #pragma omp atomic capture
918 #endif
919  {
920  oldVal = *vu;
921  *vu -= decr;
922  }
923  }
924 
925  if(oldVal == -1) {
926  // First task to touch this saddle, compute the valence
927  idVertex totalVal = starVect.size();
928  valence newVal = 0;
929  if(localProp->goUp()) {
930  valence *const vd = &graph_.valDown_[curSaddle];
931 #ifdef TTK_ENABLE_OPENMP
932 #pragma omp atomic capture
933 #endif
934  {
935  newVal = *vd;
936  *vd += (totalVal + 1);
937  }
938 
939  } else {
940  valence *const vu = &graph_.valUp_[curSaddle];
941 #ifdef TTK_ENABLE_OPENMP
942 #pragma omp atomic capture
943 #endif
944  {
945  newVal = *vu;
946  *vu += (totalVal + 1);
947  }
948  }
949  oldVal = decr + newVal + (totalVal + 1);
950  }
951 
952 #if defined(__GNUC__)
953 #pragma GCC diagnostic pop
954 #endif // __GNUC__
955 
956  return oldVal == decr;
957  }
