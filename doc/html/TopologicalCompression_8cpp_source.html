404 {

406 
407  int numberOfBytesWritten = 0;
408 
409  // Compute number of bits per segment
410  // (can be deduced at read-time from numberOfSegments)
411  int numberOfBitsPerSegment = log2(numberOfSegments) + 1;
412 
413  // [MEDIUM] TODO: support long int
414  if (numberOfBitsPerSegment > 32) return -3;
415 
416  // Encode
417  int currentCell = 0;
418  int offset = 0;
419  int maskerRank = 0;
420  while (currentCell < numberOfVertices) {
421 
422  // Create next container.
423  int compressedInt = 0;
424 
425  // Write regularly all segments until one overlaps
426  // two containers.
427  while (offset + numberOfBitsPerSegment <= 32) {
428 
429  int currentSegment = segmentation[currentCell];
430 
431  // If applicable, fill last part of current segment.
432  if (maskerRank != 0) {
433  // Always a positive int.
434  currentSegment = currentSegment >> (numberOfBitsPerSegment - offset);
435  // (numberOfBitsPerSegment - maskerRank - 1);
436  maskerRank = 0;
437  compressedInt |= currentSegment;
438 
439  } else {
440  int cursor = (currentSegment << offset); // 0es after <<
441  compressedInt |= cursor;
442  offset += numberOfBitsPerSegment;
443  }
444 
445  currentCell++; // Next cell.
446 
447  }
448 
449  // Test for overflow filling last part of current container.
450  if (currentCell >= numberOfVertices) {
451  numberOfBytesWritten += sizeof(int);
452  WriteInt(fm, compressedInt);
453  break;
454  }
455 
456  // Write current segment into last part of current container,
457  // to be continued into next container.
458  {
459  int currentSegment = segmentation[currentCell];
460 
461  if (offset == 32) {
462  // currentCell++;
463  } else {
464  int cursor = (currentSegment << offset);
465  compressedInt = compressedInt | cursor;
466 
467  maskerRank = 32 - offset;
468  offset += numberOfBitsPerSegment;
469  }
470 
471  offset %= 32;
472  }
473 
474  // Dump current container.
475  numberOfBytesWritten += sizeof(int);

477  }
