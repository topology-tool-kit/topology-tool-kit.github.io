193 {

195  if (segmentation_.size()) {
196  cout << "createSegmentation called on an already segmented region" << endl;
197  }
198 #endif
199 
200  SimplexId totalSegmSize = 0;
201  vector<segm_const_it> heads, ends;
202  for (const auto& region : segmentsIn_) {
203  totalSegmSize += distance(region.segmentBegin, region.segmentEnd);
204  heads.emplace_back(region.segmentBegin);
205  ends.emplace_back(region.segmentEnd);
206  }
207 
208  segmentation_.clear();
209  segmentation_.reserve(totalSegmSize); // max size, including discarded vertices
210 
211  idSegment nbSegments = heads.size();
212  int added = 0;
213 
214  while (added != -1) {
215  added = -1;
216  SimplexId minVert;
217  for (idSegment i = 0; i < nbSegments; i++) {
218  auto& headIt = heads[i];
219  const auto& endIt = ends[i];
220 
221  if (headIt == endIt) {
222  // end of this area
223  heads.erase(heads.begin() + i);
224  ends.erase(ends.begin() + i);
225  --nbSegments;
226  --i;
227  continue;
228  }
229 
230  if (added == -1 || s->isLower(*headIt, minVert)) {
231  minVert = *headIt;
232  added = i;
233  }
234  }
235  if (added != -1) {
236  segmentation_.emplace_back(minVert);
237  ++heads[added];
238  }
239  } // end while
