590int ExplicitTriangulation::preconditionDistributedCells() {
591 if(this->hasPreconditionedDistributedCells_) {
592 return 0;
593 }
594 if(!ttk::isRunningWithMPI()) {
595 return -1;
596 }
597 if(this->cellGid_ == nullptr) {
598 this->printWrn("Missing global identifiers on cells");
599 return -2;
600 }
601 if(this->cellRankArray_ == nullptr) {
602 this->printWrn("Missing RankArray on cells");
603 return -3;
604 }
605
606 Timer tm{};
607
608 // number of local cells (with ghost cells...)
609 const auto nLocCells{this->getNumberOfCells()};
610
611 // global cell id -> local cell id (reverse of this->cellGid_)
612 this->cellGidToLid_.reserve(nLocCells);
613 for(LongSimplexId lcid = 0; lcid < nLocCells; ++lcid) {
614 this->cellGidToLid_[this->cellGid_[lcid]] = lcid;
615 }
616
617 this->ghostCellPerOwner_.resize(ttk::MPIsize_);
618
619 for(LongSimplexId lcid = 0; lcid < nLocCells; ++lcid) {
620 if(this->cellRankArray_[lcid] != ttk::MPIrank_) {
621 // store ghost cell global ids (per rank)
622 this->ghostCellPerOwner_[this->cellRankArray_[lcid]].emplace_back(
623 this->cellGid_[lcid]);
624 }
625 }
626
627 // store neighboring ranks
628 for(size_t i = 0; i < this->ghostCellPerOwner_.size(); ++i) {
629 if(!this->ghostCellPerOwner_[i].empty()) {
630 this->neighborRanks_.emplace_back(i);
631 }
632 }
633
634 // for each rank, store the global id of local cells that are ghost cells of
635 // other ranks.
636 const auto MIT{ttk::getMPIType(ttk::SimplexId{})};
637 this->remoteGhostCells_.resize(ttk::MPIsize_);
638 // number of owned cells that are ghost cells of other ranks
639 std::vector<size_t> nOwnedGhostCellsPerRank(ttk::MPIsize_);
640
641 for(const auto neigh : this->neighborRanks_) {
642 // 1. send to neigh number of ghost cells owned by neigh
643 const auto nCells{this->ghostCellPerOwner_[neigh].size()};
644 MPI_Sendrecv(&nCells, 1, ttk::getMPIType(nCells), neigh, ttk::MPIrank_,
645 &nOwnedGhostCellsPerRank[neigh], 1, ttk::getMPIType(nCells),
646 neigh, neigh, ttk::MPIcomm_, MPI_STATUS_IGNORE);
647 this->remoteGhostCells_[neigh].resize(nOwnedGhostCellsPerRank[neigh]);
648
649 // 2. send to neigh list of ghost cells owned by neigh
650 MPI_Sendrecv(this->ghostCellPerOwner_[neigh].data(),
651 this->ghostCellPerOwner_[neigh].size(), MIT, neigh,
652 ttk::MPIrank_, this->remoteGhostCells_[neigh].data(),
653 this->remoteGhostCells_[neigh].size(), MIT, neigh, neigh,
654 ttk::MPIcomm_, MPI_STATUS_IGNORE);
655 }
656
657 this->preconditionDistributedCellRanges();
658
659 this->hasPreconditionedDistributedCells_ = true;
660
661 if(ttk::MPIrank_ == 0) {
662 this->printMsg("Domain contains "
663 + std::to_string(this->gatheredCellRanges_.back().end + 1)
664 + " cells",
665 1.0, tm.getElapsedTime(), this->threadNumber_);
666 }
667
668 return 0;
669}

671int ttk::ExplicitTriangulation::preconditionDistributedCellRanges() {
672
673 // 1. store all local cells owned by current rank by global id
674
675 std::vector<SimplexId> localCellIds{};
676 localCellIds.reserve(this->getNumberOfCells());
677 for(SimplexId i = 0; i < this->getNumberOfCells(); ++i) {
678 if(this->cellRankArray_[i] == ttk::MPIrank_) {
679 localCellIds.emplace_back(i);
680 }
681 }
682
683 TTK_PSORT(this->threadNumber_, localCellIds.begin(), localCellIds.end(),
684 [this](const SimplexId a, const SimplexId b) {
685 return this->cellGid_[a] < this->cellGid_[b];
686 });
687
688 // 2. determine ranges of contiguous cell global ids
689
690 size_t begRange{};
691 while(begRange < localCellIds.size()) {
692 size_t endRange{begRange + 1};
693
694 if(begRange < localCellIds.size() - 1) {
695 for(size_t j = begRange + 1; j < localCellIds.size(); ++j) {
696 if(this->cellGid_[localCellIds[j]]
697 > this->cellGid_[localCellIds[j - 1]] + 1) {
698 endRange = j;
699 break;
700 }
701 }
702 if(endRange == begRange + 1
703 && this->cellGid_[localCellIds[endRange]]
704 == this->cellGid_[localCellIds[endRange - 1]] + 1) {
705 endRange = localCellIds.size();
706 }
707 }
708
709 const size_t gbeg = this->cellGid_[localCellIds[begRange]];
710 const size_t gend = this->cellGid_[localCellIds[endRange - 1]];
711 const auto nRanges{this->localCellRanges_.size()};
712
713 // inclusive range
714 this->localCellRanges_.emplace_back(
715 CellRange{nRanges, gbeg, gend, static_cast<size_t>(ttk::MPIrank_)});
716
717 begRange = endRange;
718 }
719
720 // 3. send to rank 0 the vector of ranges so it can compute range offsets
721
722 if(ttk::MPIrank_ == 0) {
723 this->nRangesPerRank_.resize(ttk::MPIsize_);
724 }
725
726 const int rangeSize = this->localCellRanges_.size();
727 MPI_Gather(&rangeSize, 1, ttk::getMPIType(rangeSize),
728 this->nRangesPerRank_.data(), 1, ttk::getMPIType(rangeSize), 0,
729 ttk::MPIcomm_);
730
731 std::vector<int> displacements{};
732
733 if(ttk::MPIrank_ == 0) {
734 const auto nRanges{std::accumulate(
735 this->nRangesPerRank_.begin(), this->nRangesPerRank_.end(), 0)};
736 this->gatheredCellRanges_.resize(nRanges);
737 displacements.resize(this->nRangesPerRank_.size());
738
739 for(size_t i = 0; i < this->nRangesPerRank_.size() - 1; ++i) {
740 displacements[i + 1] = displacements[i] + this->nRangesPerRank_[i];
741 }
742 }
743
744 auto cellRangeDT{CellRange::getMPIType()};
745 MPI_Type_commit(&cellRangeDT);
746
747 MPI_Gatherv(this->localCellRanges_.data(), this->localCellRanges_.size(),
748 cellRangeDT, this->gatheredCellRanges_.data(),
749 this->nRangesPerRank_.data(), displacements.data(), cellRangeDT,
750 0, ttk::MPIcomm_);
751
752 MPI_Type_free(&cellRangeDT);
753
754 // 4. sort range vector on rank 0
755
756 if(ttk::MPIrank_ == 0) {
757 TTK_PSORT(
758 this->threadNumber_, this->gatheredCellRanges_.begin(),
759 this->gatheredCellRanges_.end(),
760 [](const CellRange &a, const CellRange &b) { return a.begin < b.begin; });
761 }
762
763 return 0;
764}

772
773 if(ttk::MPIrank_ == 0) {
774 nSimplicesPerRangePerRank.resize(this->nRangesPerRank_.size());
775 for(int i = 0; i < ttk::MPIsize_; ++i) {
776 nSimplicesPerRangePerRank[i].resize(this->nRangesPerRank_[i]);
777 }
778 for(int i = 0; i < ttk::MPIsize_; ++i) {
779 if(i == 0) {
780 continue;
781 }
782 // receive src content from other ranks
783 MPI_Recv(nSimplicesPerRangePerRank[i].data(),
784 nSimplicesPerRangePerRank[i].size(), ttk::getMPIType(size_t{}),
785 i, MPI_ANY_TAG, ttk::MPIcomm_, MPI_STATUS_IGNORE);
786 }
787 std::swap(nSimplicesPerRangePerRank[0], nSimplicesPerRange);
788 } else {
789 MPI_Send(nSimplicesPerRange.data(), nSimplicesPerRange.size(),
790 ttk::getMPIType(size_t{}), 0, 0, ttk::MPIcomm_);
791 }
792
793 // 2. compute range offsets on rank 0
794
795 size_t nSimplices{};
796 if(ttk::MPIrank_ == 0) {
797
798 for(const auto &range : this->gatheredCellRanges_) {
799 auto &pSum{nSimplicesPerRangePerRank[range.rank][range.id]};
800 std::swap(pSum, nSimplices);
801 nSimplices += pSum;
802 }
803 }
804
805 // 3. send back range offsets to other ranks
806
807 if(ttk::MPIrank_ == 0) {
808 for(int i = 1; i < ttk::MPIsize_; ++i) {
809 MPI_Send(nSimplicesPerRangePerRank[i].data(),
810 nSimplicesPerRangePerRank[i].size(), ttk::getMPIType(size_t{}),
811 i, 0, ttk::MPIcomm_);
812 }
813 std::swap(nSimplicesPerRange, nSimplicesPerRangePerRank[0]);
814 } else {
815 MPI_Recv(nSimplicesPerRange.data(), nSimplicesPerRange.size(),
816 ttk::getMPIType(size_t{}), MPI_ANY_TAG, 0, ttk::MPIcomm_,
817 MPI_STATUS_IGNORE);
818 }
819
820 return nSimplices;
821}

823template <typename Func0, typename Func1, typename Func2>
824int ttk::ExplicitTriangulation::exchangeDistributedInternal(
825 const Func0 &getGlobalSimplexId,
826 const Func1 &storeGlobalSimplexId,
827 const Func2 &iterCond,
828 const int nSimplicesPerCell) {
829
830 // per neighbor, owned ghost cell simplex global ids to transfer back
831 std::vector<std::vector<SimplexId>> globalIdPerOwnedGhostCell(ttk::MPIsize_);
832 // per neighbor, non-owned ghost cell simplex global ids to transfer back
833 std::vector<std::vector<SimplexId>> globalIdPerLocalGhostCell(ttk::MPIsize_);
834
835 const auto MIT{ttk::getMPIType(ttk::SimplexId{})};
836
837 // make sure that all simplices are correctly labelled: for a given
838 // rank, a simplex can be owned by a ghost cell from a neighboring
839 // rank but in reality can be owned by another ghost cell in a third
840 // rank
841 bool doIter{true};
842
843 while(doIter) {
844
845 doIter = false;
846
847 // 3. for each list of ghost cell, accumulate the global simplex id
848 for(const auto neigh : this->neighborRanks_) {
849 // sending side
850 globalIdPerOwnedGhostCell[neigh].resize(
851 nSimplicesPerCell * this->remoteGhostCells_[neigh].size());
852 for(size_t i = 0; i < this->remoteGhostCells_[neigh].size(); ++i) {
853 const auto lcid{this->cellGidToLid_[this->remoteGhostCells_[neigh][i]]};
854 for(int j = 0; j < nSimplicesPerCell; ++j) {
855 globalIdPerOwnedGhostCell[neigh][nSimplicesPerCell * i + j]
856 = getGlobalSimplexId(lcid, j);
857 }
858 }
859 // receiving side
860 globalIdPerLocalGhostCell[neigh].resize(
861 nSimplicesPerCell * this->ghostCellPerOwner_[neigh].size());
862
863 // 4. transfer back global simplex ids
864 MPI_Sendrecv(globalIdPerOwnedGhostCell[neigh].data(),
865 globalIdPerOwnedGhostCell[neigh].size(), MIT, neigh,
866 ttk::MPIrank_, globalIdPerLocalGhostCell[neigh].data(),
867 globalIdPerLocalGhostCell[neigh].size(), MIT, neigh, neigh,
868 ttk::MPIcomm_, MPI_STATUS_IGNORE);
869 }
870
871 // 5. extend local <-> global simplex ids mappings
872 for(const auto neigh : this->neighborRanks_) {
873 for(size_t i = 0; i < this->ghostCellPerOwner_[neigh].size(); ++i) {
874 const auto gcid{this->ghostCellPerOwner_[neigh][i]};
875 const auto lcid{this->cellGidToLid_[gcid]};
876 for(int j = 0; j < nSimplicesPerCell; ++j) {
877 const auto geid{
878 globalIdPerLocalGhostCell[neigh][nSimplicesPerCell * i + j]};
879 storeGlobalSimplexId(lcid, geid, j);
880 }
881 }
882 }
883
884 // do an additional transmission if there still is some locally
885 // non-labelled simplices
886 int doNextIter{0};
887 if(iterCond()) {
888 doNextIter = 1;
889 doIter = true;
890 }
891 for(int i = 0; i < ttk::MPIsize_; ++i) {
892 if(doIter) {
893 // reset doNextIter (might have been erased by the MPI_Bcast)
894 doNextIter = 1;
895 }
896 MPI_Bcast(&doNextIter, 1, ttk::getMPIType(doNextIter), i, ttk::MPIcomm_);
897 doIter |= (doNextIter == 1);
898 }
899
900 if(doIter && ttk::MPIrank_ == 0) {
901 this->printMsg("Re-sending global ids to neighbors...");
902 }
903 }
904
905 return 0;
906}

908int ExplicitTriangulation::preconditionDistributedEdges() {
909 if(this->hasPreconditionedDistributedEdges_) {
910 return 0;
911 }
912 if(!ttk::isRunningWithMPI()) {
913 return -1;
914 }
915 if(this->cellGid_ == nullptr) {
916 this->printWrn("Missing global identifiers on cells");
917 return -2;
918 }
919
920 if(this->getDimensionality() != 2 && this->getDimensionality() != 3) {
921 return -3;
922 }
923
924 Timer tm{};
925
926 this->preconditionDistributedCells();
927
928 // allocate memory
929 this->edgeLidToGid_.resize(this->getNumberOfEdgesInternal(), -1);
930 this->edgeGidToLid_.reserve(this->getNumberOfEdgesInternal());
931
932 // 1. for every range of local cells, number the edges locally
933
934 std::vector<SimplexId> edgeLidToRangeId(this->getNumberOfEdgesInternal(), -1);
935 std::vector<size_t> nEdgesPerRange(this->localCellRanges_.size());
936
937 const auto edgeAlreadyProcessed = [this](const SimplexId leid,
938 const SimplexId lcid) {
939 const auto nStar{this->TTK_TRIANGULATION_INTERNAL(getEdgeStarNumber)(leid)};
940 for(SimplexId i = 0; i < nStar; ++i) {
941 SimplexId sid{-1};
942 this->TTK_TRIANGULATION_INTERNAL(getEdgeStar)(leid, i, sid);
943 if(sid == -1 || sid == lcid) {
944 continue;
945 }
946 // rule: an edge is owned by the cell in its star with the
947 // lowest global id
948 if(this->cellGid_[sid] < this->cellGid_[lcid]) {
949 return true;
950 break;
951 }
952 }
953 return false;
954 };
955
956 const auto countCellEdges
957 = [this, &edgeAlreadyProcessed](const SimplexId lcid,
958 std::vector<SimplexId> &edgeGid,
959 std::vector<SimplexId> &edgeRangeId,
960 const size_t rangeId, size_t &edgeCount) {
961 SimplexId nEdges{};
962 if(this->maxCellDim_ == 3) {
963 nEdges = this->getCellEdgeNumberInternal(lcid);
964 } else if(this->maxCellDim_ == 2) {
965 nEdges = this->getTriangleEdgeNumberInternal(lcid);
966 }
967 for(SimplexId k = 0; k < nEdges; ++k) {
968 SimplexId leid{-1};
969 if(this->maxCellDim_ == 3) {
970 this->getCellEdgeInternal(lcid, k, leid);
971 } else if(this->maxCellDim_ == 2) {
972 this->getTriangleEdge(lcid, k, leid);
973 }
974 const auto alreadyProcessed = edgeAlreadyProcessed(leid, lcid);
975 if(!alreadyProcessed) {
976 edgeGid[leid] = edgeCount;
977 edgeRangeId[leid] = rangeId;
978 edgeCount++;
979 }
980 }
981 };
982
983#ifdef TTK_ENABLE_OPENMP
984#pragma omp parallel for num_threads(this->threadNumber_)
985#endif // TTK_ENABLE_OPENMP
986 for(size_t i = 0; i < this->localCellRanges_.size(); ++i) {
987 auto &range{this->localCellRanges_[i]};
988 range.id = i;
989 for(size_t j = range.begin; j <= range.end; ++j) {
990 // local cell id
991 const auto lcid{this->cellGidToLid_[j]};
992 countCellEdges(lcid, this->edgeLidToGid_, edgeLidToRangeId, range.id,
993 nEdgesPerRange[i]);
994 }
995 }
996
997 // 2. compute range offset on rank 0
998 const auto nEdges = this->computeCellRangeOffsets(nEdgesPerRange);
999
1000 // 3. locally edit the edge global id with range offsets
1001
1002 for(SimplexId leid = 0; leid < this->getNumberOfEdgesInternal(); ++leid) {
1003 if(this->edgeLidToGid_[leid] == -1) {
1004 // not owned by a cell of this rank
1005 continue;
1006 }
1007 const auto geid{this->edgeLidToGid_[leid]
1008 + nEdgesPerRange[edgeLidToRangeId[leid]]};
1009 this->edgeLidToGid_[leid] = geid;
1010 this->edgeGidToLid_[geid] = leid;
1011 }
1012
1013 // 4. exchange global ids between ghost cells
1014
1015 const auto nEdgesPerCell{this->getDimensionality() == 3 ? 6 : 3};
1016 this->exchangeDistributedInternal(
1017 [this](const SimplexId lcid, const int j) {
1018 SimplexId leid{};
1019 this->getCellEdgeInternal(lcid, j, leid);
1020 return this->edgeLidToGid_[leid];
1021 },
1022 [this](const SimplexId lcid, const SimplexId geid, const int j) {
1023 SimplexId leid{};
1024 this->getCellEdgeInternal(lcid, j, leid);
1025 if(this->edgeLidToGid_[leid] == -1 && geid != -1) {
1026 this->edgeLidToGid_[leid] = geid;
1027 this->edgeGidToLid_[geid] = leid;
1028 }
1029 },
1030 [this]() {
1031 return std::count(
1032 this->edgeLidToGid_.begin(), this->edgeLidToGid_.end(), -1)
1033 > 0;
1034 },
1035 nEdgesPerCell);
1036
1037 if(MPIrank_ == 0) {
1038 this->printMsg("Domain contains " + std::to_string(nEdges) + " edges", 1.0,
1039 tm.getElapsedTime(), 1);
1040 }
1041
1042 this->hasPreconditionedDistributedEdges_ = true;
1043
1044 return 0;
1045}

1048 if(this->hasPreconditionedDistributedTriangles_) {
1049 return 0;
1050 }

1053 }

1071 // 1. for every range of local cells, number the edges locally

1101 size_t &triangleCount) {

1113 };

1115#ifdef TTK_ENABLE_OPENMP
1116#pragma omp parallel for num_threads(this->threadNumber_)
1117#endif // TTK_ENABLE_OPENMP
1118 for(size_t i = 0; i < this->localCellRanges_.size(); ++i) {
1119 auto &range{this->localCellRanges_[i]};
1120 range.id = i;
1121 for(size_t j = range.begin; j <= range.end; ++j) {
1122 // local cell id
1123 const auto lcid{this->cellGidToLid_[j]};
1124 countCellTriangles(lcid, this->triangleLidToGid_, triangleLidToRangeId,
1125 range.id, nTrianglesPerRange[i]);
1126 }
1127 }
1128
1129 // 2. compute range offset on rank 0
1130 const auto nTriangles = this->computeCellRangeOffsets(nTrianglesPerRange);
1131
1132 // 3. locally edit the triangle global id with range offsets
1133
1134 for(SimplexId leid = 0; leid < this->getNumberOfTrianglesInternal(); ++leid) {
1135 if(this->triangleLidToGid_[leid] == -1) {
1136 // not owned by a cell of this rank
1137 continue;
1138 }
1139 const auto geid{this->triangleLidToGid_[leid]
1140 + nTrianglesPerRange[triangleLidToRangeId[leid]]};
1141 this->triangleLidToGid_[leid] = geid;
1142 this->triangleGidToLid_[geid] = leid;

1149 [this](const SimplexId lcid, const int j) {
1150 SimplexId ltid{};
1151 this->getCellTriangleInternal(lcid, j, ltid);
1152 return this->triangleLidToGid_[ltid];
1153 },
1154 [this](const SimplexId lcid, const SimplexId gtid, const int j) {
1155 SimplexId ltid{};
1156 this->getCellTriangleInternal(lcid, j, ltid);
1157 if(this->triangleLidToGid_[ltid] == -1 && gtid != -1) {
1158 this->triangleLidToGid_[ltid] = gtid;
1159 this->triangleGidToLid_[gtid] = ltid;
1160 }
1161 },
1162 [this]() {
1163 return std::count(this->triangleLidToGid_.begin(),
1164 this->triangleLidToGid_.end(), -1)
1165 > 0;
1166 },
1167 nTrianglesPerCell);
1168
1169 if(MPIrank_ == 0) {
1170 this->printMsg(
1171 "Domain contains " + std::to_string(nTriangles) + " triangles", 1.0,
1172 tm.getElapsedTime(), 1);
1173 }
1174
1175 this->hasPreconditionedDistributedTriangles_ = true;
1176
1177 return 0;
1178}
